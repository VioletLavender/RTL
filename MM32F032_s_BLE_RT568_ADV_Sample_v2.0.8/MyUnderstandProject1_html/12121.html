106             u16                         UART_WordLength;

108         union {
109             UART_Stop_Bits_TypeDef      StopBits;                           ///< Specifies the number of stop bits transmitted.
110             u16                         UART_StopBits;
111         };
112         union {
113             UART_Parity_TypeDef         Parity;                             ///< Specifies the parity mode.
114             u16                         UART_Parity;
115         };
116         union {
117             u16                         Mode;                               ///< Specifies wether the Receive or Transmit mode is
118             u16                         UART_Mode;
119         };
120         union {
121             UART_HW_FLOWCONTROL_TypeDef HWFlowControl;                      ///< Specifies wether the hardware flow control mode is enabled or disabled.
122             u16                         UART_HardwareFlowControl;
123         };
124     } UART_InitTypeDef;
