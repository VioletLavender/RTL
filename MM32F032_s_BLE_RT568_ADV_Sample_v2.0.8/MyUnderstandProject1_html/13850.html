37     ////////////////////////////////////////////////////////////////////////////////
38     /// @brief  Deinitializes the PWR peripheral registers to their default reset
39     /// values.
40     /// @param  None.
41     /// @retval None.
42     ////////////////////////////////////////////////////////////////////////////////
43     void PWR_DeInit(void)
44     {
45         exRCC_APB1PeriphReset(RCC_APB1ENR_PWR);
46     }
47     
48     ////////////////////////////////////////////////////////////////////////////////
49     /// @brief  Enables or disables access to the RTC and backup registers.
50     /// @param  state: new state of the access to the RTC and backup
51     ///         registers. This parameter can be: ENABLE or DISABLE.
52     /// @retval None.
53     ////////////////////////////////////////////////////////////////////////////////
54     void PWR_BackupAccessCmd(FunctionalState state)
55     {
56         (state) ? (PWR->CR |= PWR_CR_DBP) : (PWR->CR &= ~PWR_CR_DBP);
57     }
58     
59     ////////////////////////////////////////////////////////////////////////////////
60     /// @brief  Enables or disables the Power Voltage Detector(PVD).
61     /// @param  state: new state of the PVD.
62     ///         This parameter can be: ENABLE or DISABLE.
63     /// @retval None.
64     ////////////////////////////////////////////////////////////////////////////////
65     void PWR_PVDCmd(FunctionalState state)
66     {
67         (state) ? (PWR->CR |= PWR_CR_PVDE) : (PWR->CR &= ~PWR_CR_PVDE);
68     }

70     ////////////////////////////////////////////////////////////////////////////////
71     /// @brief  Configures the voltage threshold detected by the Power Voltage
72     ///         Detector(PVD).
73     /// @param  PWR_PVDLevel: specifies the PVD detection level
74     ///         This parameter can be one of the following values:
75     /// @arg    PWR_PVDLevel_1V8: PVD detection level set to 2.8V
76     /// @arg    PWR_PVDLevel_2V1: PVD detection level set to 2.1V
77     /// @arg    PWR_PVDLevel_2V4: PVD detection level set to 2.4V
78     /// @arg    PWR_PVDLevel_2V7: PVD detection level set to 2.7V
79     /// @arg    PWR_PVDLevel_3V0: PVD detection level set to 3.0V
80     /// @arg    PWR_PVDLevel_3V3: PVD detection level set to 3.3V
81     /// @arg    PWR_PVDLevel_3V6: PVD detection level set to 3.6V
82     /// @arg    PWR_PVDLevel_3V9: PVD detection level set to 3.9V
83     /// @arg    PWR_PVDLevel_4V2: PVD detection level set to 4.2V
84     /// @arg    PWR_PVDLevel_4V5: PVD detection level set to 4.5V
85     /// @arg    PWR_PVDLevel_4V8: PVD detection level set to 4.8V
86     /// @retval None.
87     ////////////////////////////////////////////////////////////////////////////////
88     void PWR_PVDLevelConfig(u32 PWR_PVDLevel)
89     {
90         PWR->CR = (PWR->CR & (~PWR_CR_PLS)) | PWR_PVDLevel;
91     }

93     ////////////////////////////////////////////////////////////////////////////////
94     /// @brief  Enables or disables the WakeUp Pin functionality.
95     /// @param  state: new state of the WakeUp Pin functionality.
96     ///         This parameter can be: ENABLE or DISABLE.
97     /// @retval None.
98     ////////////////////////////////////////////////////////////////////////////////
99     void PWR_WakeUpPinCmd(FunctionalState state)
100     {
101         (state != DISABLE) ? (PWR->CSR |= PWR_CSR_EWUP) : (PWR->CSR &= ~PWR_CSR_EWUP);
102     }

104     ////////////////////////////////////////////////////////////////////////////////
105     /// @brief  Enters STOP mode.
106     /// @param  PWR_Regulator: specifies the regulator state in STOP mode.
107     ///         This parameter can be one of the following values:
108     /// @arg    PWR_Regulator_ON: STOP mode with regulator ON
109     /// @arg    PWR_Regulator_LowPower: STOP mode with regulator in low power mode.
110     /// @param  PWR_STOPEntry: specifies if STOP mode in entered with WFI or  WFE
111     ///         instruction.
112     ///         This parameter can be one of the following values:
113     /// @arg    PWR_STOPEntry_WFI: enter STOP mode with WFI instruction
114     /// @arg    PWR_STOPEntry_WFE: enter STOP mode with WFE instruction
115     /// @retval None.
116     ////////////////////////////////////////////////////////////////////////////////
117     void PWR_EnterSTOPMode(u32 PWR_Regulator, u8 PWR_STOPEntry)
118     {
119         PWR->CR |= PWR_Regulator;
120     
121         SCB->SCR |= SCB_SCR_SLEEPDEEP_Msk;
122     
123         (PWR_STOPEntry == PWR_STOPEntry_WFI) ? __WFI() : __WFE();
124     }
125     
126     ////////////////////////////////////////////////////////////////////////////////
127     /// @brief  Enters STANDBY mode.
128     /// @param  None.
129     /// @retval None.
130     ////////////////////////////////////////////////////////////////////////////////
131     void PWR_EnterSTANDBYMode(void)
132     {
133         PWR->CR |= PWR_CR_CWUF | PWR_CR_PDDS;
134         SCB->SCR |= SCB_SCR_SLEEPDEEP_Msk;
135     
136     #if defined(__CC_ARM)
137         __force_stores();
138     #endif
139         __WFI();
140     }

142     ////////////////////////////////////////////////////////////////////////////////
143     /// @brief  Checks whether the specified PWR flag is set or not.
144     /// @param  flag: specifies the flag to check.
145     ///         This parameter can be one of the following values:
146     /// @arg    PWR_FLAG_WU: Wake Up flag
147     /// @arg    PWR_FLAG_SB: StandBy flag
148     /// @arg    PWR_FLAG_PVDO: PVD Output
149     /// @retval The new state of PWR_FLAG (SET or RESET).
150     ////////////////////////////////////////////////////////////////////////////////
151     FlagStatus PWR_GetFlagStatus(u32 flag)
152     {
153         return (FlagStatus)(PWR->CSR & flag);
154     }
155     
156     ////////////////////////////////////////////////////////////////////////////////
157     /// @brief  Clears the PWR's pending flags.
158     /// @param  flag: specifies the flag to clear.
159     ///         This parameter can be one of the following values:
160     /// @arg    PWR_FLAG_WU: Wake Up flag
161     /// @arg    PWR_FLAG_SB: StandBy flag
162     /// @retval None.
163     ////////////////////////////////////////////////////////////////////////////////
164     void PWR_ClearFlag(u32 flag)
165     {
166         PWR->CR |= flag << 2;
167     }

169     ////////////////////////////////////////////////////////////////////////////////
170     /// @brief  Enters STANDBY mode.
171     /// @param  None.
172     /// @retval None.
173     ////////////////////////////////////////////////////////////////////////////////
174     void exPWR_EnterLowPowerMode(emPWR_LP_Mode_Typedef lp_Mode, emPWR_Wait_Mode_Typedef wait_Mode)
175     {
176         if (lp_Mode != LP_SLEEP_MODE) {
177             PWR->CR |= lp_Mode | PWR_CR_CWUF;
178             SCB->SCR |= SCB_SCR_SLEEPDEEP_Msk;
179             if (wait_Mode == LP_WFE) {
180                 __WFE();        //  stop & standby
181                 return;
182             }
183         }
184         __WFI();        // sleep &  stop & standby
185     
186     }
