161     {

202     void spiGpioWriteReg(const unsigned char regAddr, const unsigned char regData)
203     {
204     
205         unsigned char SPICount;                               // Counter used to clock out the data
206         unsigned char SPIData;                                // Define a data structure for the SPI data.
207     
208         SPI_CS = 1;                                           // Make sure we start with /CS high
209         SPI_CK = 0;                                           // and CK low
210     
211         SPI_CS = 0;                                           // Set /CS low to start the SPI cycle 25nS
212         // Although SPIData could be implemented as an "int", resulting in one
213         // loop, the routines run faster when two loops are implemented with
214         // SPIData implemented as two "char"s.
215     
216         delay_us(1);
217     
218         //Address 1th byte
219         SPIData = regAddr & 0x7F;
220         for (SPICount = 0; SPICount < 8; SPICount++)          // Prepare to clock out the Address byte
221         {
222             if (SPIData & 0x80)                                 // Check for a 1
223                 SPI_MOSI = 1;                                     // and set the MOSI line appropriately
224             else
225                 SPI_MOSI = 0;
226     
227             delay_us(1);                                    // delay half clk cycle
228             SPI_CK = 1;                                         // Toggle the clock line
229             delay_us(1);
230             SPI_CK = 0;
231             SPIData <<= 1;                                      // Rotate to get the next bit
232         }                                                     // and loop back to send the next bit
233         // Repeat for the Data byte
234         //Address 2nd byte
235         SPIData = (regAddr & 0x80) >> 7;
236         for (SPICount = 0; SPICount < 8; SPICount++)          // Prepare to clock out the Address byte
237         {
238             if (SPIData & 0x80)                                 // Check for a 1
239                 SPI_MOSI = 1;                                     // and set the MOSI line appropriately
240             else
241                 SPI_MOSI = 0;
242     
243             delay_us(1);                                     // delay half clk cycle
244             SPI_CK = 1;                                         // Toggle the clock line
245             delay_us(1);
246             SPI_CK = 0;
247             SPIData <<= 1;                                      // Rotate to get the next bit
248         }
249     
250         //Data
251         SPIData = regData;                                    // Preload the data to be sent with Data
252         for (SPICount = 0; SPICount < 8; SPICount++)          // Prepare to clock out the Data
253         {
254             if (SPIData & 0x80)
255                 SPI_MOSI = 1;
256             else
257                 SPI_MOSI = 0;
258     
259             delay_us(1);
260             SPI_CK = 1;
261             delay_us(1);
262             SPI_CK = 0;
263             SPIData <<= 1;
264         }
265     
266         delay_us(1);
267     
268         SPI_CS = 1;
269         SPI_MOSI = 0;
270     }

278     void SPI_GPIO_Init(void)
279     {
280         GPIO_InitTypeDef GPIO_InitStructure;
281     
282         /* GPIOB Peripheral clock enable */
283         RCC_AHBPeriphClockCmd(RCC_AHBPeriph_GPIOB | RF_IRQ_CLOCK, ENABLE);
284         /*Be careful not to set 5 pins as HIGH, it will trigger HW mapping SPI pin*/
285         GPIO_StructInit(&GPIO_InitStructure);
286         GPIO_InitStructure.GPIO_Pin   = RF_IRQ_PIN | GPIO_Pin_12 | GPIO_Pin_13 |  
287                                         GPIO_Pin_14 | GPIO_Pin_15;
288         GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
289         GPIO_InitStructure.GPIO_Mode  = GPIO_Mode_Out_PP;
290         GPIO_Init(GPIOB, &GPIO_InitStructure);
291         GPIO_Init(RF_IRQ_PORT, &GPIO_InitStructure);
292     }

294     /******************************************************************************
295      * @brief
296      * @param
297      * @retval
298      * @attention  SPI IO mapping. Must do this after Power ON
299     ******************************************************************************/
300     void RF_SpiIoMapping(void)
301     {
302         /*
303          *------------------------------------------------
304          *          NSS     SCK     MISO    MOSI   IRQ  AF
305          *------------------------------------------------
306          *   SPI2   PB12    PB13    PB14    PB15   PB8  AF0
307          *------------------------------------------------
308          */
309         /*(1) Here set all CLK,CS,MOSI,NISO,INT as output */
310         SPI_GPIO_Init();
311         SPI_CS = 1;
312         SPI_CK = 0;
313         SPI_MOSI = 0;
314         SPI_MISO = 0;
315         DEFAULT_INT = 0;
316     
317         //(2) Write R248, R249 GPIO select
318         //Write R248=8'b10,001,000, R249=8'b0,100,011,0
319         //GPIO0[2:0]=0 - CS
320         //GPIO1[2:0]=1 - CLK
321         //GPIO2[2:0]=2 - MOSI
322         //GPIO3[2:0]=3 - MISO
323         //GPIO4[2:0]=4 - INT
324         delay_us(25);
325         spiGpioWriteReg(248, 0x88);
326         delay_us(25);
327         spiGpioWriteReg(249, 0x46);
328         delay_us(25);
329     
330         /* (3) Output all pin as HIGH, last 10ms(>1ms). trigger HW take effect */
331         SPI_GPIO_Init();
332         SPI_CS = 1;
333         SPI_CK = 1;
334         SPI_MOSI = 1;
335         SPI_MISO = 1;
336         DEFAULT_INT = 1;
337         SysTick_DelayMs(10);  //Delay 10ms
338     
339         //(4) Init GPIO & SPI
340         MCU_GpioPinInit();        //Set GPIO interrupt pin as input
341         /* Init SPI_BLE : SPI master, clk=12M, mode 0, 8-bit, MSB first  */
342         BSP_SPI_BLE_Init();
343     
344         SPI_1BYT_SetTx(249, (0x46 | 0x80));  //set RF MISO, INT as output
345     
346         SPI_1BYT_SetTx(53, 0xC0); //To gurantee DC/DC power on when set R40=0x90
347     
348         //enable LDO
349         SPI_1BYT_SetTx(40, 0xC0);// 截取正常波形中是 0x90 而不是 0xc0
350         SysTick_DelayMs(25);        //Put delay after LDO_enable, or set register may have strange behavior!
351     
352         //enable chip
353         SPI_1BYT_SetTx(53, 0x80); //Enable chip
354         SysTick_DelayMs(10);   //Put delay after chip_enable, or set register may have strange behavior!
355     }

363     static void RF_Open()
364     {    
365         GPIO_InitTypeDef GPIO_InitStructure;
366     
367         /* Wait RF PHY stable ,delay 25ms */
368         SysTick_DelayMs(25);
369     
370         /* RF_RESET_CLOCK  enable */
371         RCC_AHBPeriphClockCmd(RF_RESET_CLOCK, ENABLE);
372         /* Configure RF_RESET_PIN in output pushpull mode */
373         GPIO_StructInit(&GPIO_InitStructure);
374         GPIO_InitStructure.GPIO_Pin   = RF_RESET_PIN;
375         GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
376         GPIO_InitStructure.GPIO_Mode  = GPIO_Mode_Out_PP;
377         GPIO_Init(RF_RESET_PORT, &GPIO_InitStructure);
378     
379         /* Reset RF by using RF_RESET_PIN */
380         GPIO_SetBits(RF_RESET_PORT, RF_RESET_PIN);
381         SysTick_DelayMs(1);
382         GPIO_ResetBits(RF_RESET_PORT, RF_RESET_PIN);
383         SysTick_DelayMs(1);
384         GPIO_SetBits(RF_RESET_PORT, RF_RESET_PIN);
385     
386     
387         SysTick_DelayMs(50);
388     
389         /* SPI IO remapping */
390         RF_SpiIoMapping();
391     
392         /* Initialize RF PHY */
393         RF_Init();       /* EnableGpioInterrupt in the end of this function */
394     }
