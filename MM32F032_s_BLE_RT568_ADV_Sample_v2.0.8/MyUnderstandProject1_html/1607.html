125     ////////////////////////////////////////////////////////////////////////////////
126     /// @brief  DMA Init structure definition
127     ////////////////////////////////////////////////////////////////////////////////
128     typedef struct {
129         u32 DMA_PeripheralBaseAddr;                                 ///< the peripheral base address for DMA Channeln.
130         u32 DMA_MemoryBaseAddr;                                     ///< the memory base address for DMA Channeln.
131         DMA_data_transfer_direction_TypeDef DMA_DIR;                ///< the peripheral is the source or destination.
132         u32 DMA_BufferSize;                                         ///< Specifies the buffer size, in data unit, of the Buffer size
133         DMA_peripheral_incremented_mode_TypeDef DMA_PeripheralInc;  ///< Specifies whether the Peripheral address increment or not
134         DMA_memory_incremented_mode_TypeDef DMA_MemoryInc;          ///< Specifies whether the memory address register is increment or not
135         DMA_peripheral_data_size_TypeDef DMA_PeripheralDataSize;    ///< Specifies the Peripheral data width.
136         DMA_memory_data_size_TypeDef DMA_MemoryDataSize;            ///< Specifies the Memory data width.
137         DMA_circular_normal_mode_TypeDef DMA_Mode;                  ///< Specifies the operation mode of the DMA Channeln  circular or normal mode.
138         DMA_priority_level_TypeDef DMA_Priority;                    ///< Specifies the software priority for the DMA priority level
139         DMA_memory_to_memory_TypeDef DMA_M2M;                       ///< Specifies if the DMA Channeln will be used in  memory-to-memory transfer.
140         DMA_auto_reload_TypeDef DMA_Auto_reload;                     ///< Specifies if the DMA Channeln will auto reload the CNDTR register
141     } DMA_InitTypeDef;
