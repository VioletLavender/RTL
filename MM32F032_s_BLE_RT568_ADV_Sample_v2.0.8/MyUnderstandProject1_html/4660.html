40     /******************************************************************************
41      * @brief       
42      * @param       
43      * @retval      
44      * @attention   
45     ******************************************************************************/
46     void bsp_irq_gpio_Init(void)
47     {
48         GPIO_InitTypeDef GPIO_InitStructure;
49     
50         /* GPIOA and GPIOB Peripheral clock enable */
51         RCC_AHBPeriphClockCmd(RCC_AHBPeriph_GPIOA | RCC_AHBPeriph_GPIOB, ENABLE);
52     
53         /* Configure PA15 in output pushpull mode */
54         GPIO_StructInit(&GPIO_InitStructure);
55         GPIO_InitStructure.GPIO_Pin   = GPIO_Pin_15;
56         GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
57         GPIO_InitStructure.GPIO_Mode  = GPIO_Mode_Out_PP;
58         GPIO_Init(GPIOA, &GPIO_InitStructure);
59     
60         /* Reset PA15 : turn LED1 on */
61         GPIO_ResetBits(GPIOA, GPIO_Pin_15);
62     
63         /* Configure PB3 PB4 and PB5 in output pushpull mode */
64         GPIO_StructInit(&GPIO_InitStructure);
65         GPIO_InitStructure.GPIO_Pin   = GPIO_Pin_3 | GPIO_Pin_4 | GPIO_Pin_5;
66         GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
67         GPIO_InitStructure.GPIO_Mode  = GPIO_Mode_Out_PP;
68         GPIO_Init(GPIOB, &GPIO_InitStructure);
69     
70         /* Reset PB3 PB4 and PB5 : turn LED2 LED3 and LED4 on */
71         GPIO_ResetBits(GPIOB, GPIO_Pin_3 | GPIO_Pin_4 | GPIO_Pin_5);
72     }

75     /******************************************************************************
76      * @brief    Assign GPIO INT pin
77      * @param       
78      * @retval      
79      * @attention  
80     ******************************************************************************/
81     void MCU_GpioPinInit(void)
82     {
83         GPIO_InitTypeDef GPIO_InitStructure;
84         
85         /* RF_IRQ_CLOCK enable */
86         RCC_AHBPeriphClockCmd(RF_IRQ_CLOCK, ENABLE);
87         /*Be careful not to set 5 pins as HIGH, it will trigger HW mapping SPI pin*/
88         GPIO_StructInit(&GPIO_InitStructure);
89         GPIO_InitStructure.GPIO_Pin   = RF_IRQ_PIN ;
90         GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
91         GPIO_InitStructure.GPIO_Mode  = GPIO_Mode_FLOATING; //GPIO_Mode_IPU
92         GPIO_Init(RF_IRQ_PORT, &GPIO_InitStructure);  
93     }

95     void MCU_GpioIntEnable(void)
96     {
97         NVIC_InitTypeDef  NVIC_InitStructure;
98     	EXTI_InitTypeDef  EXTI_InitStructure;
99     	
100         // Configure PF.15 as Input mode and enable interrupt by rising edge trigger
101         MCU_GpioPinInit();      
102     	RCC_APB2PeriphClockCmd(RCC_APB2Periph_SYSCFG, ENABLE);
103     	//IRQ - pb8
104     	SYSCFG_EXTILineConfig(GPIO_PortSourceGPIOB, GPIO_PinSource8);
105     
106     	EXTI_InitStructure.EXTI_Line = EXTI_Line8;
107     	EXTI_InitStructure.EXTI_Mode = EXTI_Mode_Interrupt;
108     	EXTI_InitStructure.EXTI_Trigger = EXTI_Trigger_Rising;
109     	EXTI_InitStructure.EXTI_LineCmd = ENABLE;
110     	EXTI_Init(&EXTI_InitStructure);
111     	
112     	NVIC_InitStructure.NVIC_IRQChannel = EXTI4_15_IRQn;
113     	NVIC_InitStructure.NVIC_IRQChannelPriority = 2;
114     	NVIC_InitStructure.NVIC_IRQChannelCmd = ENABLE;
115     	NVIC_Init(&NVIC_InitStructure);
116     }

120     {
121         NVIC_InitTypeDef  NVIC_InitStructure;
122     	EXTI_InitTypeDef  EXTI_InitStructure;
123     	     
124     	RCC_APB2PeriphClockCmd(RCC_APB2Periph_SYSCFG, ENABLE);
125     	//IRQ - pb8
126     	SYSCFG_EXTILineConfig(GPIO_PortSourceGPIOB, GPIO_PinSource8);
127     
128     	EXTI_InitStructure.EXTI_Line = EXTI_Line8;
129     	EXTI_InitStructure.EXTI_Mode = EXTI_Mode_Interrupt;
130     	EXTI_InitStructure.EXTI_Trigger = EXTI_Trigger_Rising;
131     	EXTI_InitStructure.EXTI_LineCmd = DISABLE;
132     	EXTI_Init(&EXTI_InitStructure);
133     	
134     	NVIC_InitStructure.NVIC_IRQChannel = EXTI4_15_IRQn;
135     	NVIC_InitStructure.NVIC_IRQChannelPriority = 2;
136     	NVIC_InitStructure.NVIC_IRQChannelCmd = DISABLE;
137     	NVIC_Init(&NVIC_InitStructure);
138     }

143     void EXTI4_15_IRQHandler(void)
144     {
145         extern void LL_GPIO_Isr(void);
146         
147         if(EXTI_GetITStatus(EXTI_Line8)!=RESET)
148         {
149             EXTI_ClearITPendingBit(EXTI_Line8);

151         LL_GPIO_Isr();      // Put this on the top following the clearing MCU GPIO Int status process.
