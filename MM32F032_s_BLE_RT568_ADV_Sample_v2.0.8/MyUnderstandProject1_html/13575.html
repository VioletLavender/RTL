174     void LL_ReleaseConnID(Uint8 LL_Conn_ID)
175     {
176         LL_Conn *pLL_conn;
177         extern Uint8 MBlk_depth_Remaining;
178         extern Uint8 LL_ConnID_Remaining;
179     
180         pLL_conn = &LL_Conn_Para[LL_Conn_ID];
181     
182         Knl_CodeCpy(&pLL_conn->LE_Conn_Para.LL_Conn_ID, TAB_ZERO_128, LEN_LE_CONN_PARA);     //initialization
183         InterruptDisable();
184         pLL_conn->LE_Conn_Para.LL_Conn_ID = LL_CONN_ID_STBY;                                 //initialization
185         pLL_conn->LE_Conn_Para.LL_Tx_PHYS = LL_TX_PHYS_1M_PHY;                               //initialization
186         pLL_conn->LE_Conn_Para.LL_Rx_PHYS = LL_RX_PHYS_1M_PHY;                               //initialization
187         pLL_conn->LE_Conn_Para.LL_Rx_length_1M = LEN_CONN_DATA_DEFAULT;                      //initialization
188         pLL_conn->LE_Conn_Para.LL_Rx_length_2M = LEN_CONN_DATA_DEFAULT;                      //initialization
189         pLL_conn->LE_Conn_Para.LL_Tx_length_1M = LEN_CONN_DATA_DEFAULT;                      //initialization
190         pLL_conn->LE_Conn_Para.LL_Tx_length_2M = LEN_CONN_DATA_DEFAULT;                      //initialization
191      
192         InterruptEnable();
193     
194     }

238     Uint8 RF_cmpFIFO_BDAddr(Uint8 HeaderSts_Rx, MBLK *pMBlk)
239     {
240         if(LL_Adv_Para.Adv_Para.LL_Own_Addr_Type == 0)
241         {
242             if((HeaderSts_Rx&FLD_MSK_LL_ADV_RXADDR_TYPE))
243             {
244                 return FAIL;
245             }
246             else
247             {
248                 return Knl_MemComp_Isr(ble_device_param.ble_deviceAddr_param.addr, &pMBlk->Para.Data[LEN_BD_ADDR], LEN_BD_ADDR);
249     
250             }
251         }
252         else
253         {
254             if((HeaderSts_Rx&FLD_MSK_LL_ADV_RXADDR_TYPE))
255             {
256                 return Knl_MemComp_Isr(BD_Rand_Addr, &pMBlk->Para.Data[LEN_BD_ADDR], LEN_BD_ADDR);
257             }
258             else
259             {
260                 return FAIL;
261             }
262         }
263     }

266     Uint8 RF_cmpFIFO_WhiteList(Uint8 HeaderSts_Rx, MBLK *pMBlk)
267     {
268         Uint8 j, status;
269     
270         j = 0;
271         if((HeaderSts_Rx&FLD_MSK_LL_ADV_TXADDR_TYPE))
272         {
273             j = HCI_ADDR_TYPE_RANDOM;
274         }
275         for(status=0; status<SIZE_WHITE_LIST_ENTRIES; status++)
276         {
277             if(LL_WhiteList_Para[status].LE_WhiteList_Para.AddrType == j)
278             {
279                 if(Knl_MemComp_Isr(pMBlk->Para.Data, &(LL_WhiteList_Para[status].LE_WhiteList_Para.Addr)[0], LEN_BD_ADDR)==SUCCESS)
280                 {
281                     break;
282                 }
283             }
284         }
285         if(status == SIZE_WHITE_LIST_ENTRIES)
286             return FAIL;
287         else
288             return SUCCESS;
289     }

292     void LL_TmrBlk_Rls(void)
293     {
294         Uint8 i;
295     
296         i = tblk_LL_pi->Next;
297         tblk_LL_pi->Next = TBlk_Free_LL;
298         TBlk_Free_LL = TBlk_InUse_LL;
299         TBlk_InUse_LL = i;
300         if(i == MAX_TBLK_LL_NO)
301         {
302             tblk_LL_pi = (TBLK_LLx *)0;
303         }
304         else
305         {
306             tblk_LL_pi = tblk_LL_pi2;
307             i = tblk_LL_pi->Next;
308         }
309         if(i == MAX_TBLK_LL_NO)
310         {
311             tblk_LL_pi2 = (TBLK_LLx *)0;
312         }
313         else
314         {
315             tblk_LL_pi2 = &TmrBlk_LL[i];
316         }
317     }

320     void LL_TmrBlk_Rls_Pair(void)
321     {
322         Uint8 i;
323         Uint16 j;
324     
325         i = tblk_LL_pi2->Next;
326         tblk_LL_pi2->Next = TBlk_Free_LL;
327         TBlk_Free_LL = TBlk_InUse_LL;
328         TBlk_InUse_LL = i;
329         j = tblk_LL_pi2->Ticks;
330     
331         if(i == MAX_TBLK_LL_NO)
332         {
333             tblk_LL_pi = (TBLK_LLx *)0;
334         }
335         else
336         {
337             tblk_LL_pi = &TmrBlk_LL[i];
338             tblk_LL_pi->Ticks = tblk_LL_pi->Ticks + j;
339             i = tblk_LL_pi->Next;
340         }
341         if(i == MAX_TBLK_LL_NO)
342         {
343             tblk_LL_pi2 = (TBLK_LLx *)0;
344         }
345         else
346         {
347             tblk_LL_pi2 = &TmrBlk_LL[i];
348         }
349     }
