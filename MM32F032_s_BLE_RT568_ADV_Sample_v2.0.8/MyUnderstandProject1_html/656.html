39     ////////////////////////////////////////////////////////////////////////////////
40     /// @brief  Resets the RCC clock configuration to default state.
41     /// @param  None.
42     /// @retval None.
43     ////////////////////////////////////////////////////////////////////////////////
44     void RCC_DeInit()
45     {
46         SET_BIT(RCC->CR, RCC_CR_HSION);
47         CLEAR_BIT(RCC->CFGR, RCC_CFGR_SW);
48         CLEAR_BIT(RCC->CR, RCC_CR_HSEON | RCC_CR_CSSON | RCC_CR_PLLON | RCC_CR_PLLDIV | RCC_CR_PLLMUL);
49         CLEAR_BIT(RCC->CR, RCC_CR_HSEBYP);
50         CLEAR_REG(RCC->CFGR);
51         CLEAR_REG(RCC->CIR);
52     }

54     ////////////////////////////////////////////////////////////////////////////////
55     /// @brief  Configures the External High Speed oscillator (HSE).
56     /// @param  state: specifies the new state of HSE.
57     ///   This parameter can be one of the following values:
58     /// @arg    RCC_HSE_OFF: HSE oscillator OFF
59     /// @arg    RCC_HSE_ON: HSE oscillator ON
60     /// @arg    RCC_HSE_Bypass: HSE oscillator bypassed with external clock
61     /// @retval None.
62     ////////////////////////////////////////////////////////////////////////////////
63     void RCC_HSEConfig(RCCHSE_TypeDef state)
64     {
65         RCC->CR &= ~(RCC_CR_HSEBYP | RCC_CR_HSEON);
66         switch (state) {
67             case RCC_HSE_Bypass:
68                 RCC->CR |= RCC_CR_HSEBYP;
69             case RCC_HSE_ON:
70                 RCC->CR |= RCC_CR_HSEON;
71             default:
72                 break;
73         }
74     }

76     ////////////////////////////////////////////////////////////////////////////////
77     /// @brief  Checks whether the specified RCC flag is set or not.
78     /// @param  flag: specifies the flag to check.
79     ///   This parameter can be one of the following values:
80     /// @arg    RCC_FLAG_HSIRDY: HSI oscillator clock ready
81     /// @arg    RCC_FLAG_HSERDY: HSE oscillator clock ready
82     /// @arg    RCC_FLAG_PLLRDY: PLL clock ready
83     /// @arg    RCC_FLAG_LSERDY: LSE oscillator clock ready
84     /// @arg    RCC_FLAG_LSIRDY: LSI oscillator clock ready
85     /// @arg    RCC_FLAG_PINRST: Pin reset
86     /// @arg    RCC_FLAG_PORRST: POR/PDR reset
87     /// @arg    RCC_FLAG_SFTRST: Software reset
88     /// @arg    RCC_FLAG_IWDGRST: Independent Watchdog reset
89     /// @arg    RCC_FLAG_WWDGRST: Window Watchdog reset
90     /// @arg    RCC_FLAG_LPWRRST: Low Power reset
91     /// @retval The new state of flag (SET or RESET).
92     ////////////////////////////////////////////////////////////////////////////////
93     FlagStatus RCC_GetFlagStatus(RCC_FLAG_TypeDef flag)
94     {
95         return ((((flag >> 5) == CR_REG_INDEX) ? RCC->CR : (((flag >> 5) == BDCR_REG_INDEX) ? RCC->BDCR : RCC->CSR)) &
96                 (1 << (flag & 0x1F)))
97                ? SET : RESET;
98     }

100     ////////////////////////////////////////////////////////////////////////////////
101     /// @brief  Waits for HSE start-up.
102     /// @param  None.
103     /// @retval An ErrorStatus enumuration value:
104     /// - SUCCESS: HSE oscillator is stable and ready to use
105     /// - ERROR: HSE oscillator not yet ready
106     ////////////////////////////////////////////////////////////////////////////////
107     ErrorStatus RCC_WaitForHSEStartUp(void)
108     {
109         u32 StartUpCounter = 0;
110     
111         FlagStatus HSEStatus;
112     
113         do {
114             HSEStatus = RCC_GetFlagStatus(RCC_FLAG_HSERDY);
115             StartUpCounter++;
116         } while ((HSEStatus == RESET) && (StartUpCounter != HSE_STARTUP_TIMEOUT));
117     
118         return (ErrorStatus)(RCC_GetFlagStatus(RCC_FLAG_HSERDY) != RESET) ? SUCCESS : ERROR;
119     }
120     
121     ////////////////////////////////////////////////////////////////////////////////
122     /// @brief  Waits for flag start-up.
123     /// @param  flag: specifies the flag to check.
124     ///   This parameter can be one of the following values:
125     /// @arg    RCC_FLAG_HSIRDY: HSI oscillator clock ready
126     /// @arg    RCC_FLAG_HSERDY: HSE oscillator clock ready
127     /// @arg    RCC_FLAG_PLLRDY: PLL clock ready
128     /// @arg    RCC_FLAG_LSERDY: LSE oscillator clock ready
129     /// @arg    RCC_FLAG_LSIRDY: LSI oscillator clock ready
130     /// @arg    RCC_FLAG_PINRST: Pin reset
131     /// @arg    RCC_FLAG_PORRST: POR/PDR reset
132     /// @arg    RCC_FLAG_SFTRST: Software reset
133     /// @arg    RCC_FLAG_IWDGRST: Independent Watchdog reset
134     /// @arg    RCC_FLAG_WWDGRST: Window Watchdog reset
135     /// @retval  An ErrorStatus enumuration value:
136     /// - SUCCESS: HSE oscillator is stable and ready to use
137     /// - ERROR: HSE oscillator not yet ready
138     ////////////////////////////////////////////////////////////////////////////////
139     ErrorStatus RCC_WaitForFlagStartUp(RCC_FLAG_TypeDef flag)
140     {
141         u32 StartUpCounter = 0;
142     
143         while (RCC_GetFlagStatus(flag) == RESET) {
144             if (StartUpCounter++ > HSE_STARTUP_TIMEOUT) {
145                 return ERROR;
146             }
147         }
148         return SUCCESS;
149     }

203     /// @brief  Configures the PLL clock source and multiplication factor.
204     ///   This function must be used only when the PLL is disabled.
205     /// @param  pllSrc: specifies the PLL entry clock source.
206     ///   This parameter can be one of the following values:
207     /// @arg RCC_HSI_Div4: HSI oscillator clock divided
208     ///   by 4 selected as PLL clock entry
209     /// @arg RCC_HSE_Div1: HSE oscillator clock selected
210     ///   as PLL clock entry
211     /// @arg RCC_HSE_Div2: HSE oscillator clock divided
212     ///   by 2 selected as PLL clock entry
213     /// @param  pllMul: specifies the PLL multiplication factor.
214     ///   This parameter can be RCC_PLLMul_x where x:[31:26][22:20]
215     /// @retval None.
216     ////////////////////////////////////////////////////////////////////////////////
217     void RCC_PLLConfig(RCC_PLLSource_TypeDef pllSrc, RCC_PLLMul_TypeDef pllMul)
218     {
219     
220     
221         u8 DNDM_Item[] = {0x07, 0x03, 0x05, 0x01, 0x07, 0x01, 0x09, 0x01,  // Frclk*8/4 ; Frclk*6/2 ; Frclk*8/2 ; Frclk*10/2;
222                           0x0B, 0x01, 0x0D, 0x01, 0x0F, 0x01, 0x11, 0x01,  // Frclk*12/2; Frclk*14/2; Frclk*16/2; Frclk*18/2;
223                           0x13, 0x01, 0x15, 0x01, 0x17, 0x01, 0x19, 0x01,  // Frclk*20/2; Frclk*22/2; Frclk*24/2; Frclk*26/2;
224                           0x1B, 0x01, 0x1D, 0x01, 0x1F, 0x01
225                          };             // Frclk*28/2; Frclk*30/2;
226         // Frclk*32/2;
227         MODIFY_REG(RCC->CFGR, (RCC_CFGR_PLLXTPRE | RCC_CFGR_PLLSRC), pllSrc);
228         RCC_PLLDMDNConfig((u32)DNDM_Item[pllMul >> 17], (u32)DNDM_Item[(pllMul >> 17) + 1]);
229     }

383     ////////////////////////////////////////////////////////////////////////////////
384     /// @brief  Returns the clock frequency of different on chip clocks.
385     /// @param  None.
386     /// @retval sysClk : System clock frequency
387     ////////////////////////////////////////////////////////////////////////////////
388     u32 RCC_GetSysClockFreq(void)
389     {
390         u32 clock, mul, div;
391         switch (RCC->CFGR & RCC_CFGR_SWS) {
392     
393             case RCC_CFGR_SWS_HSE:
394                 return HSE_VALUE;
395     
396             case RCC_CFGR_SWS_PLL:
397                 clock = READ_BIT(RCC->CFGR, RCC_CFGR_PLLSRC) ? (READ_BIT(RCC->CFGR, RCC_CFGR_PLLXTPRE) ? (HSE_VALUE >> 1) : HSE_VALUE)
398                         : HSI_VALUE_PLL_ON;
399                 mul = ((RCC->CR & (u32)RCC_CR_PLLMUL) >> RCC_CR_PLLMUL_Pos) + 1;
400                 div = ((RCC->CR & RCC_CR_PLLDIV) >> RCC_CR_PLLDIV_Pos) + 1;
401     
402                 return  clock * mul / div;
403     
404             default:
405                 return HSI_DIV6;
406         }
407     }

592     /// @param  it: specifies the RCC interrupt sources to be enabled or
593     /// disabled.
594     ///   This parameter can be any combination of the following values:
595     /// @arg RCC_IT_LSIRDY: LSI ready interrupt
596     /// @arg RCC_IT_LSERDY: LSE ready interrupt
597     /// @arg RCC_IT_HSIRDY: HSI ready interrupt
598     /// @arg RCC_IT_HSERDY: HSE ready interrupt

620     ////////////////////////////////////////////////////////////////////////////////
