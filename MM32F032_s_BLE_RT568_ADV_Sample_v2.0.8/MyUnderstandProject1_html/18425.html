175     void LL_ReleaseConnID(Uint8 LL_Conn_ID)
176     {
177         LL_Conn *pLL_conn;
178         extern Uint8 MBlk_depth_Remaining;
179         extern Uint8 LL_ConnID_Remaining;
180     
181         pLL_conn = &LL_Conn_Para[LL_Conn_ID];
182     
183         Knl_CodeCpy(&pLL_conn->LE_Conn_Para.LL_Conn_ID, TAB_ZERO_128, LEN_LE_CONN_PARA);     //initialization
184         InterruptDisable();
185         pLL_conn->LE_Conn_Para.LL_Conn_ID = LL_CONN_ID_STBY;                                 //initialization
186         pLL_conn->LE_Conn_Para.LL_Tx_PHYS = LL_TX_PHYS_1M_PHY;                               //initialization
187         pLL_conn->LE_Conn_Para.LL_Rx_PHYS = LL_RX_PHYS_1M_PHY;                               //initialization
188         pLL_conn->LE_Conn_Para.LL_Rx_length_1M = LEN_CONN_DATA_DEFAULT;                      //initialization
189         pLL_conn->LE_Conn_Para.LL_Rx_length_2M = LEN_CONN_DATA_DEFAULT;                      //initialization
190         pLL_conn->LE_Conn_Para.LL_Tx_length_1M = LEN_CONN_DATA_DEFAULT;                      //initialization
191         pLL_conn->LE_Conn_Para.LL_Tx_length_2M = LEN_CONN_DATA_DEFAULT;                      //initialization
192      
193         InterruptEnable();
194     
195     }

239     Uint8 RF_cmpFIFO_BDAddr(Uint8 HeaderSts_Rx, MBLK *pMBlk)
240     {
241         if(LL_Adv_Para.Adv_Para.LL_Own_Addr_Type == 0)
242         {
243             if((HeaderSts_Rx&FLD_MSK_LL_ADV_RXADDR_TYPE))
244             {
245                 return FAIL;
246             }
247             else
248             {
249                 return Knl_MemComp_Isr(ble_device_param.ble_deviceAddr_param.addr, &pMBlk->Para.Data[LEN_BD_ADDR], LEN_BD_ADDR);
250     
251             }
252         }
253         else
254         {
255             if((HeaderSts_Rx&FLD_MSK_LL_ADV_RXADDR_TYPE))
256             {
257                 return Knl_MemComp_Isr(BD_Rand_Addr, &pMBlk->Para.Data[LEN_BD_ADDR], LEN_BD_ADDR);
258             }
259             else
260             {
261                 return FAIL;
262             }
263         }
264     }

267     Uint8 RF_cmpFIFO_WhiteList(Uint8 HeaderSts_Rx, MBLK *pMBlk)
268     {
269         Uint8 j, status;
270     
271         j = 0;
272         if((HeaderSts_Rx&FLD_MSK_LL_ADV_TXADDR_TYPE))
273         {
274             j = HCI_ADDR_TYPE_RANDOM;
275         }
276         for(status=0; status<SIZE_WHITE_LIST_ENTRIES; status++)
277         {
278             if(LL_WhiteList_Para[status].LE_WhiteList_Para.AddrType == j)
279             {
280                 if(Knl_MemComp_Isr(pMBlk->Para.Data, &(LL_WhiteList_Para[status].LE_WhiteList_Para.Addr)[0], LEN_BD_ADDR)==SUCCESS)
281                 {
282                     break;
283                 }
284             }
285         }
286         if(status == SIZE_WHITE_LIST_ENTRIES)
287             return FAIL;
288         else
289             return SUCCESS;
290     }

293     void LL_TmrBlk_Rls(void)
294     {
295         Uint8 i;
296     
297         i = tblk_LL_pi->Next;
298         tblk_LL_pi->Next = TBlk_Free_LL;
299         TBlk_Free_LL = TBlk_InUse_LL;
300         TBlk_InUse_LL = i;
301         if(i == MAX_TBLK_LL_NO)
302         {
303             tblk_LL_pi = (TBLK_LLx *)0;
304         }
305         else
306         {
307             tblk_LL_pi = tblk_LL_pi2;
308             i = tblk_LL_pi->Next;
309         }
310         if(i == MAX_TBLK_LL_NO)
311         {
312             tblk_LL_pi2 = (TBLK_LLx *)0;
313         }
314         else
315         {
316             tblk_LL_pi2 = &TmrBlk_LL[i];
317         }
318     }

321     void LL_TmrBlk_Rls_Pair(void)
322     {
323         Uint8 i;
324         Uint16 j;
325     
326         i = tblk_LL_pi2->Next;
327         tblk_LL_pi2->Next = TBlk_Free_LL;
328         TBlk_Free_LL = TBlk_InUse_LL;
329         TBlk_InUse_LL = i;
330         j = tblk_LL_pi2->Ticks;
331     
332         if(i == MAX_TBLK_LL_NO)
333         {
334             tblk_LL_pi = (TBLK_LLx *)0;
335         }
336         else
337         {
338             tblk_LL_pi = &TmrBlk_LL[i];
339             tblk_LL_pi->Ticks = tblk_LL_pi->Ticks + j;
340             i = tblk_LL_pi->Next;
341         }
342         if(i == MAX_TBLK_LL_NO)
343         {
344             tblk_LL_pi2 = (TBLK_LLx *)0;
345         }
346         else
347         {
348             tblk_LL_pi2 = &TmrBlk_LL[i];
349         }
350     }
