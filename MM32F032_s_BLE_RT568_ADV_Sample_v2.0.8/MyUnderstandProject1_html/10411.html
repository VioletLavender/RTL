291     /** \brief  Union type to access the Application Program Status Register (APSR).
292      */
293     typedef union {
294         struct {
295     #if (__CORTEX_M != 0x07)
296             uint32_t _reserved0: 27;             /*!< bit:  0..26  Reserved                           */
297     #else
298             uint32_t _reserved0: 16;             /*!< bit:  0..15  Reserved                           */
299             uint32_t GE: 4;                      /*!< bit: 16..19  Greater than or Equal flags        */
300             uint32_t _reserved1: 7;              /*!< bit: 20..26  Reserved                           */
301     #endif
302             uint32_t Q: 1;                       /*!< bit:     27  Saturation condition flag          */
303             uint32_t V: 1;                       /*!< bit:     28  Overflow condition code flag       */
304             uint32_t C: 1;                       /*!< bit:     29  Carry condition code flag          */
305             uint32_t Z: 1;                       /*!< bit:     30  Zero condition code flag           */
306             uint32_t N: 1;                       /*!< bit:     31  Negative condition code flag       */
307         } b;                                   /*!< Structure used for bit  access                  */
308         uint32_t w;                            /*!< Type      used for word access                  */
309     } APSR_Type;
310

313      */
314     typedef union {
315         struct {
316             uint32_t ISR: 9;                     /*!< bit:  0.. 8  Exception number                   */
317             uint32_t _reserved0: 23;             /*!< bit:  9..31  Reserved                           */
318         } b;                                   /*!< Structure used for bit  access                  */
319         uint32_t w;                            /*!< Type      used for word access                  */
320     } IPSR_Type;

323     /** \brief  Union type to access the Special-Purpose Program Status Registers (xPSR).
324      */
325     typedef union {
326         struct {
327             uint32_t ISR: 9;                     /*!< bit:  0.. 8  Exception number                   */
328     #if (__CORTEX_M != 0x07)
329             uint32_t _reserved0: 15;             /*!< bit:  9..23  Reserved                           */
330     #else
331             uint32_t _reserved0: 7;              /*!< bit:  9..15  Reserved                           */
332             uint32_t GE: 4;                      /*!< bit: 16..19  Greater than or Equal flags        */
333             uint32_t _reserved1: 4;              /*!< bit: 20..23  Reserved                           */
334     #endif
335             uint32_t T: 1;                       /*!< bit:     24  Thumb bit        (read 0)          */
336             uint32_t IT: 2;                      /*!< bit: 25..26  saved IT state   (read 0)          */
337             uint32_t Q: 1;                       /*!< bit:     27  Saturation condition flag          */
338             uint32_t V: 1;                       /*!< bit:     28  Overflow condition code flag       */
339             uint32_t C: 1;                       /*!< bit:     29  Carry condition code flag          */
340             uint32_t Z: 1;                       /*!< bit:     30  Zero condition code flag           */
341             uint32_t N: 1;                       /*!< bit:     31  Negative condition code flag       */
342         } b;                                   /*!< Structure used for bit  access                  */
343         uint32_t w;                            /*!< Type      used for word access                  */

1620     /** \brief  Set Priority Grouping
1621     
1622       The function sets the priority grouping field using the required unlock sequence.
1623       The parameter PriorityGroup is assigned to the field SCB->AIRCR [10:8] PRIGROUP field.
1624       Only values from 0..7 are used.
1625       In case of a conflict between priority grouping and available
1626       priority bits (__NVIC_PRIO_BITS), the smallest possible priority group is set.
1627     
1628         \param [in]      PriorityGroup  Priority grouping field.
1629      */
1630     __STATIC_INLINE void NVIC_SetPriorityGrouping(uint32_t PriorityGroup)
1631     {
1632         uint32_t reg_value;
1633         uint32_t PriorityGroupTmp = (PriorityGroup & (uint32_t)0x07);               /* only values 0..7 are used          */
1634     
1635         reg_value  =  SCB->AIRCR;                                                   /* read old register configuration    */
1636         reg_value &= ~(SCB_AIRCR_VECTKEY_Msk | SCB_AIRCR_PRIGROUP_Msk);             /* clear bits to change               */
1637         reg_value  =  (reg_value                                 |
1638                        ((uint32_t)0x5FA << SCB_AIRCR_VECTKEY_Pos) |
1639                        (PriorityGroupTmp << 8));                                     /* Insert write key and priorty group */
1640         SCB->AIRCR =  reg_value;
1641     }
1642

1644     /** \brief  Get Priority Grouping
1645     
1646       The function reads the priority grouping field from the NVIC Interrupt Controller.
1647     
1648         \return                Priority grouping field (SCB->AIRCR [10:8] PRIGROUP field).
1649      */
1650     __STATIC_INLINE uint32_t NVIC_GetPriorityGrouping(void)
1651     {
1652         return ((SCB->AIRCR & SCB_AIRCR_PRIGROUP_Msk) >> SCB_AIRCR_PRIGROUP_Pos);   /* read priority grouping field */
1653     }

1747         if(IRQn < 0) {
1748             SCB->SHPR[((uint32_t)(IRQn) & 0xF) - 4] = ((priority << (8 - __NVIC_PRIO_BITS)) & 0xff);
1749         } /* set Priority for Cortex-M  System Interrupts */
1750         else {
1751             NVIC->IP[(uint32_t)(IRQn)]            = ((priority << (8 - __NVIC_PRIO_BITS)) & 0xff);
1752         } /* set Priority for device specific Interrupts  */
1753     }

1756     /** \brief  Get Interrupt Priority
1757     
1758         The function reads the priority of an interrupt. The interrupt
1759         number can be positive to specify an external (device specific)
1760         interrupt, or negative to specify an internal (core) interrupt.
1761     
1762     
1763         \param [in]   IRQn  Interrupt number.
1764         \return             Interrupt Priority. Value is aligned automatically to the implemented
1765                             priority bits of the microcontroller.
1766      */
1767     __STATIC_INLINE uint32_t NVIC_GetPriority(IRQn_Type IRQn)
1768     {
1769     
1770         if(IRQn < 0) {
1771             return((uint32_t)(SCB->SHPR[((uint32_t)(IRQn) & 0xF) - 4] >> (8 - __NVIC_PRIO_BITS)));
1772         } /* get priority for Cortex-M  system interrupts */
1773         else {
1774             return((uint32_t)(NVIC->IP[(uint32_t)(IRQn)]            >> (8 - __NVIC_PRIO_BITS)));
1775         } /* get priority for device specific interrupts  */
1776     }

1779     /** \brief  Encode Priority
1780     
1781         The function encodes the priority for an interrupt with the given priority group,
1782         preemptive priority value, and subpriority value.
1783         In case of a conflict between priority grouping and available
1784         priority bits (__NVIC_PRIO_BITS), the smallest possible priority group is set.
1785     
1786         \param [in]     PriorityGroup  Used priority group.
1787         \param [in]   PreemptPriority  Preemptive priority value (starting from 0).
1788         \param [in]       SubPriority  Subpriority value (starting from 0).
1789         \return                        Encoded priority. Value can be used in the function \ref NVIC_SetPriority().
1790      */
1791     __STATIC_INLINE uint32_t NVIC_EncodePriority (uint32_t PriorityGroup, uint32_t PreemptPriority, uint32_t SubPriority)
1792     {
1793         uint32_t PriorityGroupTmp = (PriorityGroup & 0x07);          /* only values 0..7 are used          */
1794         uint32_t PreemptPriorityBits;
1795         uint32_t SubPriorityBits;
1796     
1797         PreemptPriorityBits = ((7 - PriorityGroupTmp) > __NVIC_PRIO_BITS) ? __NVIC_PRIO_BITS : 7 - PriorityGroupTmp;
1798         SubPriorityBits     = ((PriorityGroupTmp + __NVIC_PRIO_BITS) < 7) ? 0 : PriorityGroupTmp - 7 + __NVIC_PRIO_BITS;
1799     
1800         return (
1801                    ((PreemptPriority & ((1 << (PreemptPriorityBits)) - 1)) << SubPriorityBits) |
1802                    ((SubPriority     & ((1 << (SubPriorityBits    )) - 1)))
1803                );
1804     }

1807     /** \brief  Decode Priority
1808     
1809         The function decodes an interrupt priority value with a given priority group to
1810         preemptive priority value and subpriority value.
1811         In case of a conflict between priority grouping and available
1812         priority bits (__NVIC_PRIO_BITS) the smallest possible priority group is set.
1813     
1814         \param [in]         Priority   Priority value, which can be retrieved with the function \ref NVIC_GetPriority().
1815         \param [in]     PriorityGroup  Used priority group.
1816         \param [out] pPreemptPriority  Preemptive priority value (starting from 0).
1817         \param [out]     pSubPriority  Subpriority value (starting from 0).
1818      */
1819     __STATIC_INLINE void NVIC_DecodePriority (uint32_t Priority, uint32_t PriorityGroup, uint32_t* pPreemptPriority, uint32_t* pSubPriority)
1820     {
1821         uint32_t PriorityGroupTmp = (PriorityGroup & 0x07);          /* only values 0..7 are used          */
1822         uint32_t PreemptPriorityBits;
1823         uint32_t SubPriorityBits;
1824     
1825         PreemptPriorityBits = ((7 - PriorityGroupTmp) > __NVIC_PRIO_BITS) ? __NVIC_PRIO_BITS : 7 - PriorityGroupTmp;
1826         SubPriorityBits     = ((PriorityGroupTmp + __NVIC_PRIO_BITS) < 7) ? 0 : PriorityGroupTmp - 7 + __NVIC_PRIO_BITS;
1827     
1828         *pPreemptPriority = (Priority >> SubPriorityBits) & ((1 << (PreemptPriorityBits)) - 1);
1829         *pSubPriority     = (Priority                   ) & ((1 << (SubPriorityBits    )) - 1);
1830     }

1833     /** \brief  System Reset
1834     
1835         The function initiates a system reset request to reset the MCU.
1836      */
1837     __STATIC_INLINE void NVIC_SystemReset(void)
1838     {
1839         __DSB();                                                     /* Ensure all outstanding memory accesses included
1840                                                                       buffered write are completed before reset */
1841         SCB->AIRCR  = ((0x5FA << SCB_AIRCR_VECTKEY_Pos)      |
1842                        (SCB->AIRCR & SCB_AIRCR_PRIGROUP_Msk) |
1843                        SCB_AIRCR_SYSRESETREQ_Msk);                   /* Keep priority group unchanged */
1844         __DSB();                                                     /* Ensure completion of memory access */
1845         while(1);                                                    /* wait until reset */
1846     }
