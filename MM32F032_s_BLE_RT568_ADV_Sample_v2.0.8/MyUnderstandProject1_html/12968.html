73     /// @brief  Initializes the TIMx Time Base Unit peripheral according to
74     ///         the specified parameters in the pInitStruct.
75     /// @param  TIMx: where x can be 1, 2, 3, 4, 5, 6, 7, 8, 14, 16 or 17 to select the TIM peripheral.
76     /// @param  pInitStruct: pointer to a TIM_TimeBaseInitTypeDef
77     ///         structure that contains the configuration information for the
78     ///         specified TIM peripheral.
79     /// @retval None.
80     ////////////////////////////////////////////////////////////////////////////////
81     void TIM_TimeBaseInit(TIM_TypeDef* TIMx, TIM_TimeBaseInitTypeDef* pInitStruct)
82     {
83         MODIFY_REG(TIMx->CR1, TIM_CR1_CKD, pInitStruct->TIM_ClockDivision);
84     
85         if ((TIMx == TIM1) || (TIMx == TIM2) || (TIMx == TIM3))
86             MODIFY_REG(TIMx->CR1, TIM_CR1_CMS | TIM_CR1_DIR, pInitStruct->TIM_CounterMode);
87     
88         if ((TIMx == TIM1) || (TIMx == TIM16) || (TIMx == TIM17))
89             SET_BIT(TIMx->RCR, pInitStruct->TIM_RepetitionCounter);
90     
91         SET_BIT(TIMx->ARR, pInitStruct->TIM_Period);
92         SET_BIT(TIMx->PSC, pInitStruct->TIM_Prescaler);
93         SET_BIT(TIMx->EGR, TIM_PSCReloadMode_Immediate);
94     }

217     /// @brief  Configures the TIM peripheral according to the specified
218     ///         parameters in the pInitStruct to measure an external PWM signal.
219     /// @param  TIMx: where x can be  1, 2, 3, 4, 5 or 8 to select the TIM peripheral.
220     /// @param  pInitStruct: pointer to a TIM_ICInitTypeDef structure that
221     ///         contains the configuration information for the specified TIM peripheral.
222     /// @retval None.
223     ////////////////////////////////////////////////////////////////////////////////
224     void TIM_PWMIConfig(TIM_TypeDef* TIMx, TIM_ICInitTypeDef* pInitStruct)
225     {
226         u16 icoppositepolarity  = TIM_ICPolarity_Rising;
227         u16 icoppositeselection = TIM_ICSelection_DirectTI;
228         icoppositepolarity = (pInitStruct->TIM_ICPolarity == TIM_ICPolarity_Rising) ? TIM_ICPolarity_Falling : TIM_ICPolarity_Rising;
229         icoppositeselection =
230             (pInitStruct->TIM_ICSelection == TIM_ICSelection_DirectTI) ? TIM_ICSelection_IndirectTI : TIM_ICSelection_DirectTI;
231         if (pInitStruct->TIM_Channel == TIM_Channel_1) {
232             TI1_Configure(TIMx, pInitStruct->TIM_ICPolarity, pInitStruct->TIM_ICSelection, pInitStruct->TIM_ICFilter);
233             TIM_SetIC1Prescaler(TIMx, pInitStruct->TIM_ICPrescaler);
234             TI2_Configure(TIMx, icoppositepolarity, icoppositeselection, pInitStruct->TIM_ICFilter);
235             TIM_SetIC2Prescaler(TIMx, pInitStruct->TIM_ICPrescaler);
236         }
237         else {
238             TI2_Configure(TIMx, pInitStruct->TIM_ICPolarity, pInitStruct->TIM_ICSelection, pInitStruct->TIM_ICFilter);
239             TIM_SetIC2Prescaler(TIMx, pInitStruct->TIM_ICPrescaler);
240             TI1_Configure(TIMx, icoppositepolarity, icoppositeselection, pInitStruct->TIM_ICFilter);
241             TIM_SetIC1Prescaler(TIMx, pInitStruct->TIM_ICPrescaler);
242         }
243     }

1090                 MODIFY_REG(TIMx->CCMR2, TIM_CCMR2_OC4M, mode << 8);
