41     ///         reset values.

43     /// @retval None.

45     void I2C_DeInit(I2C_TypeDef* I2Cn)
46     {
47         switch (*(u32*)&I2Cn) {
48             case (u32)I2C1:  // I2C1_BASE:

50                 break;
51             default:
52                 break;
53         }

56     ////////////////////////////////////////////////////////////////////////////////
57     /// @brief  Initializes the I2Cn peripheral according to the specified
58     /// parameters in the pInitStruct.
59     /// @param  I2Cn: where n can be 1 or 2 to select the I2C peripheral.
60     /// @param  pInitStruct: pointer to a I2C_InitTypeDef structure that
61     ///         contains the configuration information for the specified
62     ///         I2C peripheral.
63     /// @retval None.
64     ////////////////////////////////////////////////////////////////////////////////
65     void I2C_Init(I2C_TypeDef* I2Cn, I2C_InitTypeDef* pInitStruct)
66     {
67         u32 pclk1         = 8000000;
68         u32 minSclLowTime = 0;
69         u32 i2cPeriod     = 0;
70         u32 pclk1Period   = 0;
71     
72         RCC_ClocksTypeDef rcc_clocks;
73         I2Cn->IC_ENABLE &= ~I2C_ENR_ENABLE;
74         RCC_GetClocksFreq(&rcc_clocks);
75     
76         pclk1       = rcc_clocks.PCLK1_Frequency;
77         pclk1Period = 1000000000 / pclk1;
78         i2cPeriod   = 1000000000 / pInitStruct->I2C_ClockSpeed;
79     
80         minSclLowTime = pclk1 / pInitStruct->I2C_ClockSpeed ;
81         i2cPeriod = 82 / pclk1Period;
82     
83         if (pInitStruct->I2C_ClockSpeed <= 100000) {
84             I2Cn->IC_SS_SCL_LCNT = (minSclLowTime - 13 - i2cPeriod) / 2;
85             I2Cn->IC_SS_SCL_HCNT = (minSclLowTime - 13 - i2cPeriod - I2Cn->IC_SS_SCL_LCNT);
86         }
87         else {
88             I2Cn->IC_FS_SCL_LCNT = (minSclLowTime - 13 - i2cPeriod ) / 2 + 4;
89             I2Cn->IC_FS_SCL_HCNT = (minSclLowTime - 13 - I2Cn->IC_FS_SCL_LCNT - i2cPeriod);
90         }
91     
92         I2Cn->IC_CON &= ~(I2C_CR_EMPINT     |   
93                           I2C_CR_SLAVEDIS   |   
94                           I2C_CR_REPEN      |   
95                           I2C_CR_MASTER10   |   
96                           I2C_CR_SLAVE10    |   
97                           I2C_CR_FAST       |   
98                           I2C_CR_MASTER);
99     
100         I2Cn->IC_CON =  I2C_CR_EMPINT       |   
101                         I2C_CR_REPEN        |   
102                         ((pInitStruct->I2C_Speed) ? I2C_CR_FAST : I2C_CR_STD) | 
103                         ((pInitStruct->I2C_Mode)  ? I2C_CR_MASTER : 0x00);
104         I2Cn->IC_INTR_MASK &= INTR_MASK;
105     
106         I2Cn->IC_RX_TL = 0x00;
107         I2Cn->IC_TX_TL = 0x00;
108     }

143     void I2C_DMACmd(I2C_TypeDef* I2Cn, FunctionalState state)
144     {
145         if (state) {
146             if (I2C_DMA_DIR == TDMAE_SET)
147                 I2Cn->IC_DMA_CR |= TDMAE_SET;
148     
149             else
150                 I2Cn->IC_DMA_CR |= RDMAE_SET;
151         }
152         else
153             I2Cn->IC_DMA_CR &= ~(I2C_DMA_RXEN | I2C_DMA_TXEN);
154     }

169     /// @brief  Generates I2Cn communication STOP condition.
170     /// @param  I2Cn: where n can be 1 or 2 to select the I2C peripheral.
171     /// @param  state: new state of the I2C STOP condition generation.
172     ///         This parameter can be: ENABLE or DISABLE.
173     /// @retval None.
174     ////////////////////////////////////////////////////////////////////////////////
175     void I2C_GenerateSTOP(I2C_TypeDef* I2Cn, FunctionalState state)
176     {
177         u16 overTime = 3000;
178     
179         I2Cn->IC_ENABLE |= I2C_ENR_ABORT;
180     
181         while (I2Cn->IC_ENABLE & I2C_ENR_ABORT) {
182             if (overTime-- == 0)
183                 break;
184         }
185         I2Cn->IC_CLR_TX_ABRT;
186     }

245     void I2C_ITConfig(I2C_TypeDef* I2Cn, u16 it, FunctionalState state)
246     {
247         if (it == I2C_IT_RX_FULL)
248             I2C_ReadCmd(I2Cn);
249         (state) ? SET_BIT(I2Cn->IC_INTR_MASK, it) : CLEAR_BIT(I2Cn->IC_INTR_MASK, (u16)it);
250     }

324     ///   as parameter.
325     /// @param  I2Cn: where n can be 1 or 2 to select the I2C peripheral.
326     /// @param  event: specifies the event to be checked.
327     ///   This parameter can be one of the following values:
328     /// @arg  I2C_EVENT_RX_UNDER : Rx Buffer is empty event
329     /// @arg  I2C_EVENT_RX_OVER  : RX  Buffer Overrun event
330     /// @arg  I2C_EVENTT_RX_FULL : Rx buffer full event
331     /// @arg  I2C_EVENT_TX_OVER  : TX  Buffer Overrun event
332     /// @arg  I2C_EVENT_TX_EMPTY : TX_FIFO empty event
333     /// @arg  I2C_EVENT_RD_REQ   : I2C work as slave or master event
334     /// @arg  I2C_EVENT_TX_ABRT  : TX error event(Master mode)
335     /// @arg  I2C_EVENT_RX_DONE  : Master not ack event(slave mode)
336     /// @arg  I2C_EVENT_ACTIVITY : I2C activity event
337     /// @arg  I2C_EVENT_STOP_DET : stop condition  event
338     /// @arg  I2C_EVENT_START_DET: start condition  event
339     /// @arg  I2C_EVENT_GEN_CALL : a general call address and ack event
340     /// - SUCCESS: Last event is equal to the I2C_EVENT
341     /// - ERROR: Last event is different from the I2C_EVENT
342     ////////////////////////////////////////////////////////////////////////////////
343     ErrorStatus I2C_CheckEvent(I2C_TypeDef* I2Cn, u32 event)
344     {
345         if ((event == I2C_EVENT_RX_FULL) && (I2C_CMD_DIR == 0)) {
346             I2Cn->IC_DATA_CMD = I2C_DR_CMD;
347             I2C_CMD_DIR       = 1;
348         }
349     
350         return (ErrorStatus)((I2Cn->IC_RAW_INTR_STAT & event) == event);
351     }

379             I2C_CMD_DIR       = 1;
380         }
381         return (((I2Cn->IC_RAW_INTR_STAT & flag)) ? SET : RESET);

452     /// @brief  Clears the I2Cn interrupt pending bits.
453     /// @param  I2Cn: where n can be 1 or 2 to select the I2C peripheral.
454     /// @param  it: specifies the interrupt pending bit to clear.
455     ///   This parameter can be any combination of the following values:
456     /// @arg  I2C_IT_RX_UNDER : Rx Buffer is empty interrupt
457     /// @arg  I2C_IT_RX_OVER  : RX  Buffer Overrun interrupt
458     /// @arg  I2C_IT_RX_FULL  : Rx buffer full interrupt
459     /// @arg  I2C_IT_TX_OVER  : TX  Buffer Overrun interrupt
460     /// @arg  I2C_IT_TX_EMPTY : TX_FIFO empty interrupt
461     /// @arg  I2C_IT_RD_REQ   : I2C work as slave or master interrupt
462     /// @arg  I2C_IT_TX_ABRT  : TX error interrupt  (Master mode)
463     /// @arg  I2C_IT_RX_DONE  : Master not ack interrupt (slave mode)
464     /// @arg  I2C_IT_ACTIVITY : I2C activity interrupt
465     /// @arg  I2C_IT_STOP_DET : stop condition  interrupt
466     /// @arg  I2C_IT_START_DET: start condition  interrupt
467     /// @arg  I2C_IT_GEN_CALL : a general call address and ack interrupt
468     /// @retval None.
469     ////////////////////////////////////////////////////////////////////////////////
470     void I2C_ClearITPendingBit(I2C_TypeDef* I2Cn, u32 it)
471     {
472         if ((it & I2C_IT_RX_UNDER) == I2C_FLAG_RX_UNDER)
473             I2Cn->IC_CLR_RX_UNDER;
474         if ((it & I2C_IT_RX_OVER) == I2C_FLAG_RX_OVER)
475             I2Cn->IC_CLR_RX_OVER;
476         if ((it & I2C_IT_TX_OVER) == I2C_FLAG_TX_OVER)
477             I2Cn->IC_CLR_TX_OVER;
478         if ((it & I2C_IT_RD_REQ) == I2C_FLAG_RD_REQ)
479             I2Cn->IC_CLR_RD_REQ;
480         if ((it & I2C_IT_TX_ABRT) == I2C_FLAG_TX_ABRT)
481             I2Cn->IC_CLR_TX_ABRT;
482         if ((it & I2C_IT_RX_DONE) == I2C_FLAG_RX_DONE)
483             I2Cn->IC_CLR_RX_DONE;
484         if ((it & I2C_IT_ACTIVITY) == I2C_FLAG_ACTIVITY)
485             I2Cn->IC_CLR_ACTIVITY;
486         if ((it & I2C_IT_STOP_DET) == I2C_FLAG_STOP_DET)
487             I2Cn->IC_CLR_STOP_DET;
488         if ((it & I2C_IT_START_DET) == I2C_FLAG_START_DET)
489             I2Cn->IC_CLR_START_DET;
490         if ((it & I2C_IT_GEN_CALL) == I2C_FLAG_GEN_CALL)
491             I2Cn->IC_CLR_GEN_CALL;
492     }

500     ////////////////////////////////////////////////////////////////////////////////
501     /// @brief  Configures slave address.
502     /// @param  I2Cn: where n can be 1 or 2 to select the I2C peripheral.
503     /// @param  addr: specifies the slave address which will be transmitted
504     ///   This parameter can be one of the following values
505     /// @retval None.
506     ////////////////////////////////////////////////////////////////////////////////
507     void I2C_SendSlaveAddress(I2C_TypeDef* I2Cn, u8 addr)
508     {
509         WRITE_REG(I2Cn->IC_SAR, addr >> 1);
510     }

512     ////////////////////////////////////////////////////////////////////////////////
513     /// @brief  Enables or disables the I2C slave mode.
514     /// @param  I2Cn: where n can be 1 or 2 to select the I2C peripheral.
515     /// @param  state: new state of the specified I2C interrupts.
516     ///   This parameter can be: ENABLE or DISABLE.
517     /// @retval None.
518     ////////////////////////////////////////////////////////////////////////////////
519     void I2C_SlaveConfigure(I2C_TypeDef* I2Cn, FunctionalState state)
520     {
521         (state) ? CLEAR_BIT(I2Cn->IC_CON, I2C_CR_SLAVEDIS) : SET_BIT(I2Cn->IC_CON, I2C_CR_SLAVEDIS);
522     }

524     ////////////////////////////////////////////////////////////////////////////////
525     /// @brief  Configures the specified I2C DMA requests.
526     /// @param  I2Cn: where n can be 1 or 2 to select the I2C peripheral.
527     /// @param  dir : TDMAE_SET,RDMAE_SET
528     ///   This parameter can be any combination of the following values:
529     /// @arg  TDMAE_SET: DMA TX set
530     /// @arg  RDMAE_SET: DMA RX set
531     /// @retval None.
532     ////////////////////////////////////////////////////////////////////////////////
533     void I2C_DMAConfigure(I2C_TypeDef* I2Cn, u8 dir)
534     {
535         WRITE_REG(I2Cn->IC_DMA_CR, (2 - dir));
536     }
