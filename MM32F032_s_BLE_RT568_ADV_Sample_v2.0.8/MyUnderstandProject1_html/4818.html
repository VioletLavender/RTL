38     /// @brief  Deinitializes the ADCn peripheral registers to their default
39     ///         reset values.
40     /// @param  ADCn: where n can be 1, 2  to select the ADC peripheral.
41     /// @retval None.
42     ////////////////////////////////////////////////////////////////////////////////
43     void ADC_DeInit(ADC_TypeDef* ADCn)
44     {
45         switch (*(u32*)&ADCn) {
46             case ADC1_BASE:
47                 exRCC_APB2PeriphReset(RCC_APB2ENR_ADC1);
48                 break;
49             default:
50                 break;
51         }
52     }

54     ////////////////////////////////////////////////////////////////////////////////
55     /// @brief  Initializes the ADCn peripheral according to the specified parameters
56     ///         in the pInitStruct, Please use this function if you want to be
57     ///         compatible with older versions of the library.
58     /// @param  ADCn: where n can be 1, 2  to select the ADC peripheral.
59     /// @param  pInitStruct: pointer to an ADC_InitTypeDef structure that contains
60     ///         the configuration information for the specified ADC peripheral.
61     /// @retval None.
62     ////////////////////////////////////////////////////////////////////////////////
63     void ADC_Init(ADC_TypeDef* ADCn, ADC_InitTypeDef* pInitStruct)
64     {
65         ADCn->ADCFG &= ~(ADC_CFGR_PRE | ADC_CFGR_RSLTCTL);
66         ADCn->ADCFG |= (u32)(pInitStruct->ADC_PRESCARE) | pInitStruct->ADC_Resolution;
67     
68         ADCn->ADCR &= ~(ADC_CR_ALIGN | ADC_CR_MODE | ADC_CR_TRGSEL);
69         ADCn->ADCR |= ((u32)pInitStruct->ADC_DataAlign) | pInitStruct->ADC_ExternalTrigConv | ((u32)pInitStruct->ADC_Mode);
70     }
71     
72     ////////////////////////////////////////////////////////////////////////////////
73     /// @brief  Fills each pInitStruct member with its default value.
74     /// @param  pInitStruct : pointer to an ADC_InitTypeDef structure which will be
75     ///         initialized.
76     /// @retval None.
77     ////////////////////////////////////////////////////////////////////////////////
78     void ADC_StructInit(ADC_InitTypeDef* pInitStruct)
79     {
80         pInitStruct->ADC_Resolution         = ADC_Resolution_12b;
81         pInitStruct->ADC_PRESCARE           = ADC_PCLK2_PRESCARE_2;
82         pInitStruct->ADC_Mode               = ADC_CR_IMM;                           //ADC_Mode_Single;
83         pInitStruct->ADC_ContinuousConvMode = DISABLE;                              // useless
84         pInitStruct->ADC_ExternalTrigConv   = ADC1_ExternalTrigConv_T1_CC1;
85         pInitStruct->ADC_DataAlign          = ADC_DataAlign_Right;
86     }

146     ////////////////////////////////////////////////////////////////////////////////
147     /// @brief  Enable the selected ADC channel and configure its sample time. Please
148     ///         use this function if you want to be compatible with older versions
149     ///         of the library.
150     /// @param  ADCn: where n can be 1, 2 to select the ADC peripheral.
151     /// @param  channel: the ADC channel to configure.
152     /// @param  sampleTime: the ADC Channel n Sample time to configure.
153     /// @retval None.
154     ////////////////////////////////////////////////////////////////////////////////
155     void ADC_RegularChannelConfig(ADC_TypeDef* ADCn, u32 channel, u8 rank, u32 sampleTime)//ADCSAM_TypeDef
156     {
157         ADCn->ADCFG &= ~ADC_CFGR_SAMCTL;
158         if(sampleTime < 0x10)
159             sampleTime = sampleTime << 10;
160         ADCn->ADCFG |= sampleTime;
161         ADCn->ADCHS &= ~(1 << channel);
162         ADCn->ADCHS |=  (1 << channel);
163     
164     
165         if (channel & ADC_CHSR_CHT)
166             ADC_TempSensorVrefintCmd(ENABLE);
167         else if (channel & ADC_CHSR_CHV)
168             ADC_TempSensorVrefintCmd(ENABLE);
169     }

249     {
250         u32 tempThreshold;

361     ////////////////////////////////////////////////////////////////////////////////
362     /// @brief  Configures the ADCn any channels conversion rank and channel.
363     /// @param  ADCn: where n can be 1, 2 to select the ADC peripheral.
364     /// @param  Rank: Rank can be 0x0~0xf for the convert sequence.
365     /// @param  ADC_Channel: Configuring the target channel to be converted.
366     /// @retval None.
367     ////////////////////////////////////////////////////////////////////////////////
368     void ADC_ANY_CH_Config(ADC_TypeDef* ADCn, u8 Rank, ADCCHANNEL_TypeDef ADC_Channel)
369     {
370         Rank = Rank & 0xF;
371         if(Rank < 8) {
372             ADCn->CHANY0 &= ~(0x0F << (4 * Rank));
373             ADCn->CHANY0 |= (ADC_Channel << (4 * Rank));
374         }
375         else {
376             ADCn->CHANY1 &= ~(0x0F << (4 * (Rank - 8)));
377             ADCn->CHANY1 |= (ADC_Channel << (4 * (Rank - 8)));
378         }
379     }
