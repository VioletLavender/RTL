44     ////////////////////////////////////////////////////////////////////////////////
45     /// @brief  Deinitializes the EXTI registers to their default reset values.
46     /// @param  None.
47     /// @retval None.
48     /// @note   MEM_MODE bits are not affected by APB reset.
49     /// @note   MEM_MODE bits took the value from the user option bytes.
50     /// @note   CFGR2 register is not affected by APB reset.
51     /// @note   CLABBB configuration bits are locked when set.
52     /// @note   To unlock the configuration, perform a system reset.
53     ////////////////////////////////////////////////////////////////////////////////
54     void EXTI_DeInit()
55     {
56         u16 i;
57     
58     
59         // Set EXTI_CFGR1 register to reset value without affecting MEM_MODE bits
60         EXTI->CFGR &= EXTI_CFGR_MEMMODE;
61     
62         // Set EXTICRx registers to reset value
63         for (i = 0; i < 4; i++) {
64             EXTI->CR[i] = 0;
65         }
66     }

69     ////////////////////////////////////////////////////////////////////////////////
70     /// @brief  Configures the memory mapping at address 0x00000000.
71     /// @param  EXTI_MemoryRemap: selects the memory remapping.
72     ///          This parameter can be one of the following values:
73     ///            @arg EXTI_MemoryRemap_Flash: Main Flash memory mapped at 0x00000000
74     ///            @arg EXTI_MemoryRemap_SystemMemory: System Flash memory mapped at 0x00000000
75     ///            @arg EXTI_MemoryRemap_SRAM: Embedded SRAM mapped at 0x00000000
76     /// @retval None.
77     ////////////////////////////////////////////////////////////////////////////////
78     void EXTI_MemoryRemapConfig(u32 EXTI_MemoryRemap)
79     {
80         EXTI->CFGR = (EXTI->CFGR & ~EXTI_CFGR_MEMMODE) | EXTI_MemoryRemap;
81     }

83     ////////////////////////////////////////////////////////////////////////////////
84     /// @brief  Configure the DMA channels remapping.
85     /// @param  EXTI_DMARemap: selects the DMA channels remap. This parameter can be one of the following values:
86     ///            @arg EXTI_DMARemap_TIM17: Remap TIM17 DMA requests from channel1 to channel2
87     ///            @arg EXTI_DMARemap_TIM16: Remap TIM16 DMA requests from channel3 to channel4
88     ///            @arg EXTI_DMARemap_UART1Rx: Remap UART1 Rx DMA requests from channel3 to channel5
89     ///            @arg EXTI_DMARemap_UART1Tx: Remap UART1 Tx DMA requests from channel2 to channel4
90     ///            @arg EXTI_DMARemap_ADC1: Remap ADC1 DMA requests from channel1 to channel2
91     /// @param  state: new state of the DMA channel remapping.
92     ///         This parameter can be: ENABLE or DISABLE.
93     /// @note   When enabled, DMA channel of the selected peripheral is remapped
94     /// @note   When disabled, Default DMA channel is mapped to the selected peripheral
95     /// @note   By default TIM17 DMA requests is mapped to channel 1, use EXTI_DMAChannelRemapConfig(EXTI_DMARemap_TIM17, Enable)
96     /// to remap TIM17 DMA requests to channel 2 and use EXTI_DMAChannelRemapConfig(EXTI_DMARemap_TIM17, Disable) to map TIM17 DMA
97     /// requests to channel 1 (default mapping)
98     /// @retval None.
99     ////////////////////////////////////////////////////////////////////////////////
100     void EXTI_DMAChannelRemapConfig(u32 EXTI_DMARemap, FunctionalState state)
101     {
102         (state) ? (EXTI->CFGR |= EXTI_DMARemap) : (EXTI->CFGR &= ~EXTI_DMARemap);
103     }

106     ////////////////////////////////////////////////////////////////////////////////
107     /// @brief  Selects the GPIO pin used as EXTI Line.
108     /// @param  EXTI_PortSourceGPIOx: selects the GPIO port to be used as source for EXTI lines where x can be (A, B, C, D, E or F).
109     /// @param  EXTI_PinSourcex: specifies the EXTI line to be configured.
110     /// @note   This parameter can be EXTI_PinSourcex where x can be:
111     ///         For MCU: (0..15) for GPIOA, GPIOB, (13..15) for GPIOC and (0..1, 6..7) for GPIOD.
112     /// @retval None.
113     ////////////////////////////////////////////////////////////////////////////////
114     void EXTI_LineConfig(u8 EXTI_PortSourceGPIOx, u8 EXTI_PinSourcex)
115     {
116         EXTI->CR[EXTI_PinSourcex >> 0x02] &= ~(0x0F << (0x04 * (EXTI_PinSourcex & 0x03)));
117         EXTI->CR[EXTI_PinSourcex >> 0x02] |= ((EXTI_PortSourceGPIOx) << (0x04 * (EXTI_PinSourcex & 0x03)));
118     }

120     ////////////////////////////////////////////////////////////////////////////////
121     /// @brief  Initializes the EXTI peripheral according to the specified
122     ///         parameters in the pInitStruct.
123     /// @param  pInitStruct: pointer to a EXTI_InitTypeDef structure that
124     ///         contains the configuration information for the EXTI peripheral.
125     /// @retval None.
126     ////////////////////////////////////////////////////////////////////////////////
127     void EXTI_Init(EXTI_InitTypeDef* pInitStruct)
128     {
129         exEXTI_LineDisable(pInitStruct->EXTI_Line);
130     
131         if (pInitStruct->EXTI_LineCmd) {
132     
133             if (pInitStruct->EXTI_Mode == EXTI_Mode_Interrupt)
134                 EXTI->IMR |= pInitStruct->EXTI_Line;
135             else
136                 EXTI->EMR |= pInitStruct->EXTI_Line;
137     
138             if (pInitStruct->EXTI_Trigger == EXTI_Trigger_Rising_Falling) {
139                 EXTI->RTSR |= pInitStruct->EXTI_Line;
140                 EXTI->FTSR |= pInitStruct->EXTI_Line;                               // Rising and Faling    afio
141             }
142             else if (pInitStruct->EXTI_Trigger == EXTI_Trigger_Rising)
143                 EXTI->RTSR |= pInitStruct->EXTI_Line;
144             else
145                 EXTI->FTSR |= pInitStruct->EXTI_Line;
146         }
147     }
148     
149     ////////////////////////////////////////////////////////////////////////////////
150     /// @brief  Fills each pInitStruct member with its reset value.
151     /// @param  pInitStruct: pointer to a EXTI_InitTypeDef structure which will
152     ///         be initialized.
153     /// @retval None.
154     ////////////////////////////////////////////////////////////////////////////////
155     void EXTI_StructInit(EXTI_InitTypeDef* pInitStruct)
156     {
157         pInitStruct->EXTI_Line    = EXTI_LineNone;
158         pInitStruct->EXTI_Mode    = EXTI_Mode_Interrupt;
159         pInitStruct->EXTI_Trigger = EXTI_Trigger_Falling;
160         pInitStruct->EXTI_LineCmd = DISABLE;
161     }

176     ////////////////////////////////////////////////////////////////////////////////
177     /// @brief  Checks whether the specified EXTI line flag is set or not.
178     /// @param  EXTI_Line: specifies the EXTI line flag to check.

224     /// @param  EXTI_Line: specifies the EXTI lines to clear.
