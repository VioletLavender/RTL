228     /** \brief  Union type to access the Application Program Status Register (APSR).
229      */
230     typedef union {
231         struct {
232     #if (__CORTEX_M != 0x04)
233             uint32_t _reserved0: 27;             /*!< bit:  0..26  Reserved                           */
234     #else
235             uint32_t _reserved0: 16;             /*!< bit:  0..15  Reserved                           */
236             uint32_t GE: 4;                      /*!< bit: 16..19  Greater than or Equal flags        */
237             uint32_t _reserved1: 7;              /*!< bit: 20..26  Reserved                           */
238     #endif
239             uint32_t Q: 1;                       /*!< bit:     27  Saturation condition flag          */
240             uint32_t V: 1;                       /*!< bit:     28  Overflow condition code flag       */
241             uint32_t C: 1;                       /*!< bit:     29  Carry condition code flag          */
242             uint32_t Z: 1;                       /*!< bit:     30  Zero condition code flag           */
243             uint32_t N: 1;                       /*!< bit:     31  Negative condition code flag       */
244         } b;                                   /*!< Structure used for bit  access                  */
245         uint32_t w;                            /*!< Type      used for word access                  */
246     } APSR_Type;
247     
248     
249     /** \brief  Union type to access the Interrupt Program Status Register (IPSR).
250      */
251     typedef union {
252         struct {
253             uint32_t ISR: 9;                     /*!< bit:  0.. 8  Exception number                   */
254             uint32_t _reserved0: 23;             /*!< bit:  9..31  Reserved                           */
255         } b;                                   /*!< Structure used for bit  access                  */
256         uint32_t w;                            /*!< Type      used for word access                  */
257     } IPSR_Type;
258     
259     
260     /** \brief  Union type to access the Special-Purpose Program Status Registers (xPSR).
261      */
262     typedef union {
263         struct {
264             uint32_t ISR: 9;                     /*!< bit:  0.. 8  Exception number                   */
265     #if (__CORTEX_M != 0x04)
266             uint32_t _reserved0: 15;             /*!< bit:  9..23  Reserved                           */
267     #else
268             uint32_t _reserved0: 7;              /*!< bit:  9..15  Reserved                           */
269             uint32_t GE: 4;                      /*!< bit: 16..19  Greater than or Equal flags        */
270             uint32_t _reserved1: 4;              /*!< bit: 20..23  Reserved                           */
271     #endif
272             uint32_t T: 1;                       /*!< bit:     24  Thumb bit        (read 0)          */
273             uint32_t IT: 2;                      /*!< bit: 25..26  saved IT state   (read 0)          */
274             uint32_t Q: 1;                       /*!< bit:     27  Saturation condition flag          */
275             uint32_t V: 1;                       /*!< bit:     28  Overflow condition code flag       */
276             uint32_t C: 1;                       /*!< bit:     29  Carry condition code flag          */
277             uint32_t Z: 1;                       /*!< bit:     30  Zero condition code flag           */
278             uint32_t N: 1;                       /*!< bit:     31  Negative condition code flag       */
279         } b;                                   /*!< Structure used for bit  access                  */
280         uint32_t w;                            /*!< Type      used for word access                  */
281     } xPSR_Type;
282     
283     
284     /** \brief  Union type to access the Control Registers (CONTROL).
285      */
286     typedef union {
287         struct {
288             uint32_t nPRIV: 1;                   /*!< bit:      0  Execution privilege in Thread mode */
289             uint32_t SPSEL: 1;                   /*!< bit:      1  Stack to be used                   */
290             uint32_t FPCA: 1;                    /*!< bit:      2  FP extension active flag           */
291             uint32_t _reserved0: 29;             /*!< bit:  3..31  Reserved                           */
292         } b;                                   /*!< Structure used for bit  access                  */
293         uint32_t w;                            /*!< Type      used for word access                  */
294     } CONTROL_Type;

643     /** \ingroup  CMSIS_Core_FunctionInterface
644         \defgroup CMSIS_Core_NVICFunctions NVIC Functions
645         \brief      Functions that manage interrupts and exceptions via the NVIC.
646         @{
647      */
648     
649     /* Interrupt Priorities are WORD accessible only under ARMv6M                   */
650     /* The following MACROS handle generation of the register offset and byte masks */
651     #define _BIT_SHIFT(IRQn)         (  (((uint32_t)(IRQn)       )    &  0x03) * 8 )
652     #define _SHP_IDX(IRQn)           ( ((((uint32_t)(IRQn) & 0x0F)-8) >>    2)     )
653     #define _IP_IDX(IRQn)            (   ((uint32_t)(IRQn)            >>    2)     )
654     
655     
656     /** \brief  Enable External Interrupt
657     
658         The function enables a device-specific interrupt in the NVIC interrupt controller.
659     
660         \param [in]      IRQn  External interrupt number. Value cannot be negative.
661      */
662     __STATIC_INLINE void NVIC_EnableIRQ(IRQn_Type IRQn)
663     {
664         NVIC->ISER[0] = (1 << ((uint32_t)(IRQn) & 0x1F));
665     }
666     
667     
668     /** \brief  Disable External Interrupt
669     
670         The function disables a device-specific interrupt in the NVIC interrupt controller.
671     
672         \param [in]      IRQn  External interrupt number. Value cannot be negative.
673      */
674     __STATIC_INLINE void NVIC_DisableIRQ(IRQn_Type IRQn)
675     {
676         NVIC->ICER[0] = (1 << ((uint32_t)(IRQn) & 0x1F));
677     }
678     
679     
680     /** \brief  Get Pending Interrupt
681     
682         The function reads the pending register in the NVIC and returns the pending bit
683         for the specified interrupt.
684     
685         \param [in]      IRQn  Interrupt number.
686     
687         \return             0  Interrupt status is not pending.
688         \return             1  Interrupt status is pending.
689      */
690     __STATIC_INLINE uint32_t NVIC_GetPendingIRQ(IRQn_Type IRQn)
691     {
692         return((uint32_t) ((NVIC->ISPR[0] & (1 << ((uint32_t)(IRQn) & 0x1F))) ? 1 : 0));
693     }
694     
695     
696     /** \brief  Set Pending Interrupt
697     
698         The function sets the pending bit of an external interrupt.
699     
700         \param [in]      IRQn  Interrupt number. Value cannot be negative.
701      */
702     __STATIC_INLINE void NVIC_SetPendingIRQ(IRQn_Type IRQn)
703     {
704         NVIC->ISPR[0] = (1 << ((uint32_t)(IRQn) & 0x1F));
705     }
706     
707     
708     /** \brief  Clear Pending Interrupt
709     
710         The function clears the pending bit of an external interrupt.
711     
712         \param [in]      IRQn  External interrupt number. Value cannot be negative.
713      */
714     __STATIC_INLINE void NVIC_ClearPendingIRQ(IRQn_Type IRQn)
715     {
716         NVIC->ICPR[0] = (1 << ((uint32_t)(IRQn) & 0x1F)); /* Clear pending interrupt */
717     }
718     
719     
720     /** \brief  Set Interrupt Priority
721     
722         The function sets the priority of an interrupt.
723     
724         \note The priority cannot be set for every core interrupt.
725     
726         \param [in]      IRQn  Interrupt number.
727         \param [in]  priority  Priority to set.
728      */
729     __STATIC_INLINE void NVIC_SetPriority(IRQn_Type IRQn, uint32_t priority)
730     {
731         if(IRQn < 0) {
732             SCB->SHP[_SHP_IDX(IRQn)] = (SCB->SHP[_SHP_IDX(IRQn)] & ~(0xFF << _BIT_SHIFT(IRQn))) |
733                                        (((priority << (8 - __NVIC_PRIO_BITS)) & 0xFF) << _BIT_SHIFT(IRQn));
734         }
735         else {
736             NVIC->IP[_IP_IDX(IRQn)] = (NVIC->IP[_IP_IDX(IRQn)] & ~(0xFF << _BIT_SHIFT(IRQn))) |
737                                       (((priority << (8 - __NVIC_PRIO_BITS)) & 0xFF) << _BIT_SHIFT(IRQn));
738         }
739     }
740     
741     
742     /** \brief  Get Interrupt Priority
743     
744         The function reads the priority of an interrupt. The interrupt
745         number can be positive to specify an external (device specific)
746         interrupt, or negative to specify an internal (core) interrupt.
747     
748     
749         \param [in]   IRQn  Interrupt number.
750         \return             Interrupt Priority. Value is aligned automatically to the implemented
751                             priority bits of the microcontroller.
752      */
753     __STATIC_INLINE uint32_t NVIC_GetPriority(IRQn_Type IRQn)
754     {
755     
756         if(IRQn < 0) {
757             return((uint32_t)(((SCB->SHP[_SHP_IDX(IRQn)] >> _BIT_SHIFT(IRQn) ) & 0xFF) >> (8 - __NVIC_PRIO_BITS)));
758         } /* get priority for Cortex-M0 system interrupts */
759         else {
760             return((uint32_t)(((NVIC->IP[ _IP_IDX(IRQn)] >> _BIT_SHIFT(IRQn) ) & 0xFF) >> (8 - __NVIC_PRIO_BITS)));
761         } /* get priority for device specific interrupts  */
762     }

765     /** \brief  System Reset
766     
767         The function initiates a system reset request to reset the MCU.
768      */
769     __STATIC_INLINE void NVIC_SystemReset(void)
770     {
771         __DSB();                                                     /* Ensure all outstanding memory accesses included
772                                                                       buffered write are completed before reset */
773         SCB->AIRCR  = ((0x5FA << SCB_AIRCR_VECTKEY_Pos)      |
774                        SCB_AIRCR_SYSRESETREQ_Msk);
775         __DSB();                                                     /* Ensure completion of memory access */
776         while(1);                                                    /* wait until reset */
777     }

791     
792     /** \brief  System Tick Configuration
793     
794         The function initializes the System Timer and its interrupt, and starts the System Tick Timer.
795         Counter is in free running mode to generate periodic interrupts.
796     
797         \param [in]  ticks  Number of ticks between two interrupts.
798     
799         \return          0  Function succeeded.
800         \return          1  Function failed.
801     
802         \note     When the variable <b>__Vendor_SysTickConfig</b> is set to 1, then the
803         function <b>SysTick_Config</b> is not included. In this case, the file <b><i>device</i>.h</b>
804         must contain a vendor-specific implementation of this function.
805     
806      */
807     __STATIC_INLINE uint32_t SysTick_Config(uint32_t ticks)
808     {
809         if ((ticks - 1) > SysTick_LOAD_RELOAD_Msk)  return (1);      /* Reload value impossible */
810     
811         SysTick->LOAD  = ticks - 1;                                  /* set reload register */
812         NVIC_SetPriority (SysTick_IRQn, (1 << __NVIC_PRIO_BITS) - 1); /* set Priority for Systick Interrupt */
813         SysTick->VAL   = 0;                                          /* Load the SysTick Counter Value */
814         SysTick->CTRL  = SysTick_CTRL_CLKSOURCE_Msk |
815                          SysTick_CTRL_TICKINT_Msk   |
816                          SysTick_CTRL_ENABLE_Msk;                    /* Enable SysTick IRQ and SysTick Timer */
817         return (0);                                                  /* Function successful */
818     }
