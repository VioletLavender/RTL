199     /** \brief  Union type to access the Application Program Status Register (APSR).
200      */
201     typedef union {
202         struct {
203     #if (__CORTEX_M != 0x04)
204             uint32_t _reserved0: 27;             /*!< bit:  0..26  Reserved                           */
205     #else
206             uint32_t _reserved0: 16;             /*!< bit:  0..15  Reserved                           */
207             uint32_t GE: 4;                      /*!< bit: 16..19  Greater than or Equal flags        */
208             uint32_t _reserved1: 7;              /*!< bit: 20..26  Reserved                           */
209     #endif
210             uint32_t Q: 1;                       /*!< bit:     27  Saturation condition flag          */
211             uint32_t V: 1;                       /*!< bit:     28  Overflow condition code flag       */
212             uint32_t C: 1;                       /*!< bit:     29  Carry condition code flag          */
213             uint32_t Z: 1;                       /*!< bit:     30  Zero condition code flag           */
214             uint32_t N: 1;                       /*!< bit:     31  Negative condition code flag       */
215         } b;                                   /*!< Structure used for bit  access                  */
216         uint32_t w;                            /*!< Type      used for word access                  */
217     } APSR_Type;
218     
219     
220     /** \brief  Union type to access the Interrupt Program Status Register (IPSR).
221      */
222     typedef union {
223         struct {
224             uint32_t ISR: 9;                     /*!< bit:  0.. 8  Exception number                   */
225             uint32_t _reserved0: 23;             /*!< bit:  9..31  Reserved                           */
226         } b;                                   /*!< Structure used for bit  access                  */
227         uint32_t w;                            /*!< Type      used for word access                  */
228     } IPSR_Type;
229     
230     
231     /** \brief  Union type to access the Special-Purpose Program Status Registers (xPSR).
232      */
233     typedef union {
234         struct {
235             uint32_t ISR: 9;                     /*!< bit:  0.. 8  Exception number                   */
236     #if (__CORTEX_M != 0x04)
237             uint32_t _reserved0: 15;             /*!< bit:  9..23  Reserved                           */
238     #else
239             uint32_t _reserved0: 7;              /*!< bit:  9..15  Reserved                           */
240             uint32_t GE: 4;                      /*!< bit: 16..19  Greater than or Equal flags        */
241             uint32_t _reserved1: 4;              /*!< bit: 20..23  Reserved                           */
242     #endif
243             uint32_t T: 1;                       /*!< bit:     24  Thumb bit        (read 0)          */
244             uint32_t IT: 2;                      /*!< bit: 25..26  saved IT state   (read 0)          */
245             uint32_t Q: 1;                       /*!< bit:     27  Saturation condition flag          */
246             uint32_t V: 1;                       /*!< bit:     28  Overflow condition code flag       */
247             uint32_t C: 1;                       /*!< bit:     29  Carry condition code flag          */
248             uint32_t Z: 1;                       /*!< bit:     30  Zero condition code flag           */
249             uint32_t N: 1;                       /*!< bit:     31  Negative condition code flag       */
250         } b;                                   /*!< Structure used for bit  access                  */
251         uint32_t w;                            /*!< Type      used for word access                  */
252     } xPSR_Type;
253     
254     
255     /** \brief  Union type to access the Control Registers (CONTROL).
256      */
257     typedef union {
258         struct {
259             uint32_t nPRIV: 1;                   /*!< bit:      0  Execution privilege in Thread mode */
260             uint32_t SPSEL: 1;                   /*!< bit:      1  Stack to be used                   */
261             uint32_t FPCA: 1;                    /*!< bit:      2  FP extension active flag           */
262             uint32_t _reserved0: 29;             /*!< bit:  3..31  Reserved                           */
263         } b;                                   /*!< Structure used for bit  access                  */
264         uint32_t w;                            /*!< Type      used for word access                  */
265     } CONTROL_Type;

614     /** \ingroup  CMSIS_Core_FunctionInterface
615         \defgroup CMSIS_Core_NVICFunctions NVIC Functions
616         \brief      Functions that manage interrupts and exceptions via the NVIC.
617         @{
618      */
619     
620     /* Interrupt Priorities are WORD accessible only under ARMv6M                   */
621     /* The following MACROS handle generation of the register offset and byte masks */
622     #define _BIT_SHIFT(IRQn)         (  (((uint32_t)(IRQn)       )    &  0x03) * 8 )
623     #define _SHP_IDX(IRQn)           ( ((((uint32_t)(IRQn) & 0x0F)-8) >>    2)     )
624     #define _IP_IDX(IRQn)            (   ((uint32_t)(IRQn)            >>    2)     )
625     
626     
627     /** \brief  Enable External Interrupt
628     
629         The function enables a device-specific interrupt in the NVIC interrupt controller.
630     
631         \param [in]      IRQn  External interrupt number. Value cannot be negative.
632      */
633     __STATIC_INLINE void NVIC_EnableIRQ(IRQn_Type IRQn)
634     {
635         NVIC->ISER[0] = (1 << ((uint32_t)(IRQn) & 0x1F));
636     }
637     
638     
639     /** \brief  Disable External Interrupt
640     
641         The function disables a device-specific interrupt in the NVIC interrupt controller.
642     
643         \param [in]      IRQn  External interrupt number. Value cannot be negative.
644      */
645     __STATIC_INLINE void NVIC_DisableIRQ(IRQn_Type IRQn)
646     {
647         NVIC->ICER[0] = (1 << ((uint32_t)(IRQn) & 0x1F));
648     }
649     
650     
651     /** \brief  Get Pending Interrupt
652     
653         The function reads the pending register in the NVIC and returns the pending bit
654         for the specified interrupt.
655     
656         \param [in]      IRQn  Interrupt number.
657     
658         \return             0  Interrupt status is not pending.
659         \return             1  Interrupt status is pending.
660      */
661     __STATIC_INLINE uint32_t NVIC_GetPendingIRQ(IRQn_Type IRQn)
662     {
663         return((uint32_t) ((NVIC->ISPR[0] & (1 << ((uint32_t)(IRQn) & 0x1F))) ? 1 : 0));
664     }
665     
666     
667     /** \brief  Set Pending Interrupt
668     
669         The function sets the pending bit of an external interrupt.
670     
671         \param [in]      IRQn  Interrupt number. Value cannot be negative.
672      */
673     __STATIC_INLINE void NVIC_SetPendingIRQ(IRQn_Type IRQn)
674     {
675         NVIC->ISPR[0] = (1 << ((uint32_t)(IRQn) & 0x1F));
676     }
677     
678     
679     /** \brief  Clear Pending Interrupt
680     
681         The function clears the pending bit of an external interrupt.
682     
683         \param [in]      IRQn  External interrupt number. Value cannot be negative.
684      */
685     __STATIC_INLINE void NVIC_ClearPendingIRQ(IRQn_Type IRQn)
686     {
687         NVIC->ICPR[0] = (1 << ((uint32_t)(IRQn) & 0x1F)); /* Clear pending interrupt */
688     }
689     
690     
691     /** \brief  Set Interrupt Priority
692     
693         The function sets the priority of an interrupt.
694     
695         \note The priority cannot be set for every core interrupt.
696     
697         \param [in]      IRQn  Interrupt number.
698         \param [in]  priority  Priority to set.
699      */
700     __STATIC_INLINE void NVIC_SetPriority(IRQn_Type IRQn, uint32_t priority)
701     {
702         if(IRQn < 0) {
703             SCB->SHP[_SHP_IDX(IRQn)] = (SCB->SHP[_SHP_IDX(IRQn)] & ~(0xFF << _BIT_SHIFT(IRQn))) |
704                                        (((priority << (8 - __NVIC_PRIO_BITS)) & 0xFF) << _BIT_SHIFT(IRQn));
705         }
706         else {
707             NVIC->IP[_IP_IDX(IRQn)] = (NVIC->IP[_IP_IDX(IRQn)] & ~(0xFF << _BIT_SHIFT(IRQn))) |
708                                       (((priority << (8 - __NVIC_PRIO_BITS)) & 0xFF) << _BIT_SHIFT(IRQn));
709         }
710     }
711     
712     
713     /** \brief  Get Interrupt Priority
714     
715         The function reads the priority of an interrupt. The interrupt
716         number can be positive to specify an external (device specific)
717         interrupt, or negative to specify an internal (core) interrupt.
718     
719     
720         \param [in]   IRQn  Interrupt number.
721         \return             Interrupt Priority. Value is aligned automatically to the implemented
722                             priority bits of the microcontroller.
723      */
724     __STATIC_INLINE uint32_t NVIC_GetPriority(IRQn_Type IRQn)
725     {
726     
727         if(IRQn < 0) {
728             return((uint32_t)(((SCB->SHP[_SHP_IDX(IRQn)] >> _BIT_SHIFT(IRQn) ) & 0xFF) >> (8 - __NVIC_PRIO_BITS)));
729         } /* get priority for Cortex-M0 system interrupts */
730         else {
731             return((uint32_t)(((NVIC->IP[ _IP_IDX(IRQn)] >> _BIT_SHIFT(IRQn) ) & 0xFF) >> (8 - __NVIC_PRIO_BITS)));
732         } /* get priority for device specific interrupts  */
733     }

736     /** \brief  System Reset
737     
738         The function initiates a system reset request to reset the MCU.
739      */
740     __STATIC_INLINE void NVIC_SystemReset(void)
741     {
742         __DSB();                                                     /* Ensure all outstanding memory accesses included
743                                                                       buffered write are completed before reset */
744         SCB->AIRCR  = ((0x5FA << SCB_AIRCR_VECTKEY_Pos)      |
745                        SCB_AIRCR_SYSRESETREQ_Msk);
746         __DSB();                                                     /* Ensure completion of memory access */
747         while(1);                                                    /* wait until reset */
748     }

762     
763     /** \brief  System Tick Configuration
764     
765         The function initializes the System Timer and its interrupt, and starts the System Tick Timer.
766         Counter is in free running mode to generate periodic interrupts.
767     
768         \param [in]  ticks  Number of ticks between two interrupts.
769     
770         \return          0  Function succeeded.
771         \return          1  Function failed.
772     
773         \note     When the variable <b>__Vendor_SysTickConfig</b> is set to 1, then the
774         function <b>SysTick_Config</b> is not included. In this case, the file <b><i>device</i>.h</b>
775         must contain a vendor-specific implementation of this function.
776     
777      */
778     __STATIC_INLINE uint32_t SysTick_Config(uint32_t ticks)
779     {
780         if ((ticks - 1) > SysTick_LOAD_RELOAD_Msk)  return (1);      /* Reload value impossible */
781     
782         SysTick->LOAD  = ticks - 1;                                  /* set reload register */
783         NVIC_SetPriority (SysTick_IRQn, (1 << __NVIC_PRIO_BITS) - 1); /* set Priority for Systick Interrupt */
784         SysTick->VAL   = 0;                                          /* Load the SysTick Counter Value */
785         SysTick->CTRL  = SysTick_CTRL_CLKSOURCE_Msk |
786                          SysTick_CTRL_TICKINT_Msk   |
787                          SysTick_CTRL_ENABLE_Msk;                    /* Enable SysTick IRQ and SysTick Timer */
788         return (0);                                                  /* Function successful */
789     }
