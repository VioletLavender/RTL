43     /// @retval None.

46     {
47         switch (*(u32*)&GPIOx) {
48             case (u32)GPIOA:
49                 RCC_AHBPeriphClockCmd(RCC_AHBENR_GPIOA, DISABLE);
50                 break;
51             //COMMON_DisableIpClock(emCLOCK_GPIOA);  break;
52             case (u32)GPIOB:
53                 RCC_AHBPeriphClockCmd(RCC_AHBENR_GPIOB, DISABLE);
54                 break;
55             //COMMON_DisableIpClock(emCLOCK_GPIOB);  break;
56             case (u32)GPIOC:
57                 RCC_AHBPeriphClockCmd(RCC_AHBENR_GPIOC, DISABLE);
58                 break;
59             //COMMON_DisableIpClock(emCLOCK_GPIOC);  break;

62                 break;
63             //COMMON_DisableIpClock(emCLOCK_GPIOD);  break;

65                 break;
66         }

87     ////////////////////////////////////////////////////////////////////////////////
88     /// @brief  Initializes the GPIOx peripheral according to the specified
89     ///         parameters in the pInitStruct.
90     /// @param  GPIOx: where x can be (A..D) to select the GPIO peripheral.
91     /// @param  pInitStruct: pointer to a GPIO_InitTypeDef structure that
92     ///         contains the configuration information for the specified GPIO
93     ///         peripheral.
94     /// @retval None.
95     ////////////////////////////////////////////////////////////////////////////////
96     void GPIO_Init(GPIO_TypeDef* GPIOx, GPIO_InitTypeDef* pInitStruct)
97     {
98         u8   idx;
99         u8 i;
100         u32 tmp;
101         __IO u32* pReg;
102     
103         // 1x
104         u32 dat = pInitStruct->GPIO_Mode & 0x0F;
105         if (pInitStruct->GPIO_Mode & 0x10)
106             dat |= pInitStruct->GPIO_Speed;
107     
108         // 0x
109         pReg = &GPIOx->CRL;
110         for (i = 0; i < 8; i++) {
111             idx = i * 4;
112             if ((pInitStruct->GPIO_Pin) & (1 << i)) {
113                 *pReg = (*pReg & ~(0xF << idx)) | (dat << idx);
114             }
115         }
116     
117         pReg    = &GPIOx->CRH;
118         tmp = pInitStruct->GPIO_Pin >> 8;
119         for (i = 0; i < 8; i++) {
120             idx = i * 4;
121             if (tmp & (1 << i)) {
122                 *pReg = (*pReg & ~(0xF << idx)) | (dat << idx);
123             }
124         }
125     
126         // 2x,4x
127         if (pInitStruct->GPIO_Mode == GPIO_Mode_IPD)
128             GPIOx->BRR |= pInitStruct->GPIO_Pin;
129         else if (pInitStruct->GPIO_Mode == GPIO_Mode_IPU)
130             GPIOx->BSRR |= pInitStruct->GPIO_Pin;
131     }

146     /// @brief  Reads the input data of specified GPIO port pin.
147     /// @param  GPIOx: where x can be (A..D) to select the GPIO peripheral.

269         GPIOx->LCKR = pin;

284     ////////////////////////////////////////////////////////////////////////////////
285     /// @brief  Writes data to the specified GPIO data port.
286     /// @param  GPIOx: where x can be (A, B, C, D ) to select the GPIO peripheral.
287     /// @param  pin: specifies the pin for the Alternate function.
288     ///         This parameter can be GPIO_PinSourcex where x can be (0..15) for
289     ///         GPIOA, GPIOB, GPIOD and (0..12) for GPIOC .
290     /// @param  funcAF: selects the pin to used as Alternate function.
291     ///         This parameter can be the GPIO_AF_x where x can be (0..7).
292     /// @note   The pin should be used for Digital IP.
293     /// @retval None.
294     ////////////////////////////////////////////////////////////////////////////////
295     void GPIO_PinAFConfig(GPIO_TypeDef* GPIOx, u8 pin, u8 funcAF)
296     {
297         u8 shift = (pin & 0x07) * 4;
298         u32* ptr = (pin < 8) ? (u32*)&GPIOx->AFRL : (u32*)&GPIOx->AFRH;
299         *ptr = (*ptr & ~(0x0F << shift)) | (funcAF << shift);
300     }

302     ////////////////////////////////////////////////////////////////////////////////
303     /// @brief  Set the remap function and AF function of the GPIO pin.
304     /// @param  GPIOx: where x can be (A, B, C, D ) to select the GPIO peripheral.
305     /// @param  pin: specifies the pin for the Alternate function.
306     ///         This parameter can be GPIO_Pin_x where x can be (0..15) for
307     ///         GPIOA, GPIOB, GPIOD and (0..12) for GPIOC .
308     /// @param  remap: selects the pin to remap.
309     /// @param  funcAF: selects the pin to used as Alternate function.
310     ///         This parameter can be the GPIO_AF_x where x can be (0..7).
311     /// @note   The pin should be used for Digital IP.
312     /// @retval None.
313     ////////////////////////////////////////////////////////////////////////////////
314     void exGPIO_PinAFConfig(GPIO_TypeDef* GPIOx, u16 pin, s32 remap, s8 funcAF)
315     {
316         u8 i;
317         u8 shift;
318         u32* ptr;
319     
320         if (funcAF >= 0) {
321             for (i = 0; i < 32; i++) {
322                 if (pin & 0x01) {
323                     pin = i;
324                     break;
325                 }
326                 pin >>= 1;
327             }
328     
329             shift = (pin & 0x07) * 4;
330             ptr = (pin < 8) ? (u32*)&GPIOx->AFRL : (u32*)&GPIOx->AFRH;
331             *ptr = (*ptr & ~(0x0F << shift)) | (funcAF << shift);
332         }
333     }
