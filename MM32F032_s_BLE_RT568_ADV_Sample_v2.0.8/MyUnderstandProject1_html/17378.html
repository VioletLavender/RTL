166     void spiGpioWriteReg(const unsigned char regAddr, const unsigned char regData)
167     {
168     
169         unsigned char SPICount;                               // Counter used to clock out the data
170         unsigned char SPIData;                                // Define a data structure for the SPI data.
171     
172         SPI_CS = 1;                                           // Make sure we start with /CS high
173         SPI_CK = 0;                                           // and CK low
174     
175         SPI_CS = 0;                                           // Set /CS low to start the SPI cycle 25nS
176         // Although SPIData could be implemented as an "int", resulting in one
177         // loop, the routines run faster when two loops are implemented with
178         // SPIData implemented as two "char"s.
179     
180         delay_us(1);
181     
182         //Address 1th byte
183         SPIData = regAddr & 0x7F;
184         for (SPICount = 0; SPICount < 8; SPICount++)          // Prepare to clock out the Address byte
185         {
186             if (SPIData & 0x80)                                 // Check for a 1
187                 SPI_MOSI = 1;                                     // and set the MOSI line appropriately
188             else
189                 SPI_MOSI = 0;
190     
191             delay_us(1);                                    // delay half clk cycle
192             SPI_CK = 1;                                         // Toggle the clock line
193             delay_us(1);
194             SPI_CK = 0;
195             SPIData <<= 1;                                      // Rotate to get the next bit
196         }                                                     // and loop back to send the next bit
197         // Repeat for the Data byte
198         //Address 2nd byte
199         SPIData = (regAddr & 0x80)>>7;
200         for (SPICount = 0; SPICount < 8; SPICount++)          // Prepare to clock out the Address byte
201         {
202             if (SPIData & 0x80)                                 // Check for a 1
203                 SPI_MOSI = 1;                                     // and set the MOSI line appropriately
204             else
205                 SPI_MOSI = 0;
206     
207             delay_us(1);                                     // delay half clk cycle
208             SPI_CK = 1;                                         // Toggle the clock line
209             delay_us(1);
210             SPI_CK = 0;
211             SPIData <<= 1;                                      // Rotate to get the next bit
212         }
213     
214         //Data
215         SPIData = regData;                                    // Preload the data to be sent with Data
216         for (SPICount = 0; SPICount < 8; SPICount++)          // Prepare to clock out the Data
217         {
218             if (SPIData & 0x80)
219                 SPI_MOSI = 1;
220             else
221                 SPI_MOSI = 0;
222     
223             delay_us(1);
224             SPI_CK = 1;
225             delay_us(1);
226             SPI_CK = 0;
227             SPIData <<= 1;
228         }
229     
230         delay_us(1);
231     
232         SPI_CS = 1;
233         SPI_MOSI = 0;
234     }

242     void SPI_GPIO_Init(void)
243     {
244         GPIO_InitTypeDef GPIO_InitStructure;
245         
246         /* GPIOB Peripheral clock enable */
247         RCC_AHBPeriphClockCmd(RCC_AHBPeriph_GPIOB | RF_IRQ_CLOCK, ENABLE);
248         /*Be careful not to set 5 pins as HIGH, it will trigger HW mapping SPI pin*/
249         GPIO_StructInit(&GPIO_InitStructure);
250         GPIO_InitStructure.GPIO_Pin   = RF_IRQ_PIN | GPIO_Pin_12 | GPIO_Pin_13 |  
251                                             GPIO_Pin_14|GPIO_Pin_15;
252         GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
253         GPIO_InitStructure.GPIO_Mode  = GPIO_Mode_Out_PP;
254         GPIO_Init(GPIOB, &GPIO_InitStructure);
255         GPIO_Init(RF_IRQ_PORT, &GPIO_InitStructure);  
256     }

258     /******************************************************************************
259      * @brief       
260      * @param       
261      * @retval      
262      * @attention  SPI IO mapping. Must do this after Power ON 
263     ******************************************************************************/
264     void RF_SpiIoMapping(void)
265     {
266     /*
267      *------------------------------------------------
268      *          NSS     SCK     MISO    MOSI   IRQ  AF
269      *------------------------------------------------
270      *   SPI2   PB12    PB13    PB14    PB15   PB8  AF0
271      *------------------------------------------------
272      */
273         /*(1) Here set all CLK,CS,MOSI,NISO,INT as output */    
274         SPI_GPIO_Init();
275         SPI_CS=1;
276         SPI_CK=0;
277         SPI_MOSI=0;
278         SPI_MISO=0;
279         DEFAULT_INT=0;
280     //    GPIO_SetBits(GPIOB, GPIO_Pin_12);  /* SPI_CS = 1 ;*/
281     //    GPIO_ResetBits(GPIOB, GPIO_Pin_13); /* SPI_CK=0; */
282     //    GPIO_ResetBits(GPIOB, GPIO_Pin_15); /* SPI_MOSI=0; */
283     //    GPIO_ResetBits(GPIOB, GPIO_Pin_14); /* SPI_MISO=0; */
284     //    GPIO_ResetBits(RF_IRQ_PORT, RF_IRQ_PIN); /* DEFAULT_INT=0; */
285         
286         //(2) Write R248, R249 GPIO select
287         //Write R248=8'b10,001,000, R249=8'b0,100,011,0
288         //GPIO0[2:0]=0 - CS
289         //GPIO1[2:0]=1 - CLK
290         //GPIO2[2:0]=2 - MOSI
291         //GPIO3[2:0]=3 - MISO
292         //GPIO4[2:0]=4 - INT
293         SysTick_DelayMs(1);
294         spiGpioWriteReg(248, 0x88);
295         SysTick_DelayMs(1);
296         spiGpioWriteReg(249, 0x46);
297         SysTick_DelayMs(1);
298     
299         /* (3) Output all pin as HIGH, last 10ms(>1ms). trigger HW take effect */
300         SPI_GPIO_Init();
301         SPI_CS=1;
302         SPI_CK=1;
303         SPI_MOSI=1;
304         SPI_MISO=1;
305         DEFAULT_INT=1;
306         SysTick_DelayMs(10);  //Delay 10ms
307         
308         //(4) Init GPIO & SPI
309         MCU_GpioPinInit();        //Set GPIO interrupt pin as input
310         /* Init SPI_BLE : SPI master, clk=12M, mode 0, 8-bit, MSB first  */
311         BSP_SPI_BLE_Init();
312     
313         SPI_1BYT_SetTx(249, (0x46 | 0x80));  //set RF MISO, INT as output
314     
315         //manual control
316         SPI_1BYT_SetTx(53, 0xC0); //To gurantee DC/DC power on when set R40=0x90
317     
318         //enable LDO
319     //    SPI_1BYT_SetTx(40, 0xC0);  
320         SPI_1BYT_SetTx(40, 0x90);// 截取正常波形中是 0x90 而不是 0xc0
321         SysTick_DelayMs(25);        //Put delay after LDO_enable, or set register may have strange behavior!
322     
323         //enable chip
324         SPI_1BYT_SetTx(53, 0x80); //Enable chip
325         SysTick_DelayMs(10);   //Put delay after chip_enable, or set register may have strange behavior!
326     }

334     static void RF_Open()
335     { 
336         GPIO_InitTypeDef GPIO_InitStructure;
337        
338         /* Wait RF PHY stable ,delay 25ms */
339         SysTick_DelayMs(25);
340         
341         /* RF_RESET_CLOCK  enable */
342         RCC_AHBPeriphClockCmd(RF_RESET_CLOCK, ENABLE);
343         /* Configure RF_RESET_PIN in output pushpull mode */
344         GPIO_StructInit(&GPIO_InitStructure);
345         GPIO_InitStructure.GPIO_Pin   = RF_RESET_PIN;
346         GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
347         GPIO_InitStructure.GPIO_Mode  = GPIO_Mode_Out_PP;
348         GPIO_Init(RF_RESET_PORT, &GPIO_InitStructure);
349        
350         /* Reset RF by using RF_RESET_PIN */  
351         GPIO_SetBits(RF_RESET_PORT, RF_RESET_PIN);
352         SysTick_DelayMs(1);
353         GPIO_ResetBits(RF_RESET_PORT, RF_RESET_PIN);
354         SysTick_DelayMs(1);
355         GPIO_SetBits(RF_RESET_PORT, RF_RESET_PIN);
356        
357     	
358         SysTick_DelayMs(50); 
359     
360         /* SPI IO remapping */
361         RF_SpiIoMapping();
362     
363         /* Initialize RF PHY */
364         RF_Init();                   //EnableGpioInterrupt in the end of this function  
365     }
