41     ////////////////////////////////////////////////////////////////////////////////
42     /// @brief  EXTI mode enumeration
43     ////////////////////////////////////////////////////////////////////////////////
44     typedef enum {
45         EXTI_Mode_Interrupt = 0x00,                                                 ///< EXTI interrupt mode
46         EXTI_Mode_Event     = 0x04                                                  ///< EXTI event mode
47     } EXTIMode_TypeDef;
48     
49     ////////////////////////////////////////////////////////////////////////////////
50     /// @brief  EXTI Trigger enumeration
51     ////////////////////////////////////////////////////////////////////////////////
52     typedef enum {
53         EXTI_Trigger_Rising,                                                        ///< EXTI rising edge triggering
54         EXTI_Trigger_Falling,                                                       ///< EXTI falling edge triggering
55         EXTI_Trigger_Rising_Falling                                                 ///< EXTI rising and falling edge triggers
56     } EXTITrigger_TypeDef;
57     
58     ////////////////////////////////////////////////////////////////////////////////
59     /// @brief  EXTI Init Structure definition
60     ////////////////////////////////////////////////////////////////////////////////
61     typedef struct {
62         u32 EXTI_Line;                                                              ///< Specifies the EXTI lines to be enabled or disabled.
63         ///< This parameter can be any combination of @ref EXTI_Lines
64         EXTIMode_TypeDef EXTI_Mode;                                                 ///< Specifies the mode for the EXTI lines.
65         ///< This parameter can be a value of @ref EXTIMode_TypeDef
66         EXTITrigger_TypeDef EXTI_Trigger;                                           ///< Specifies the trigger signal active edge for the EXTI lines.
67         ///< This parameter can be a value of @ref EXTIMode_TypeDef
68         FunctionalState EXTI_LineCmd;                                               ///< Specifies the new state of the selected EXTI lines.
69         ///< This parameter can be set either to ENABLE or DISABLE
70     } EXTI_InitTypeDef;
