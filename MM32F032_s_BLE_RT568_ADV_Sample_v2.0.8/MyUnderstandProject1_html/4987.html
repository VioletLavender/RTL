238             uint32_t _reserved1: 7;              /*!< bit: 20..26  Reserved                           */
239     #endif
240             uint32_t Q: 1;                       /*!< bit:     27  Saturation condition flag          */
241             uint32_t V: 1;                       /*!< bit:     28  Overflow condition code flag       */
242             uint32_t C: 1;                       /*!< bit:     29  Carry condition code flag          */
243             uint32_t Z: 1;                       /*!< bit:     30  Zero condition code flag           */
244             uint32_t N: 1;                       /*!< bit:     31  Negative condition code flag       */
245         } b;                                   /*!< Structure used for bit  access                  */
246         uint32_t w;                            /*!< Type      used for word access                  */
247     } APSR_Type;

1263     /** \brief  Set Priority Grouping
1264     
1265       The function sets the priority grouping field using the required unlock sequence.
1266       The parameter PriorityGroup is assigned to the field SCB->AIRCR [10:8] PRIGROUP field.
1267       Only values from 0..7 are used.
1268       In case of a conflict between priority grouping and available
1269       priority bits (__NVIC_PRIO_BITS), the smallest possible priority group is set.
1270     
1271         \param [in]      PriorityGroup  Priority grouping field.
1272      */
1273     __STATIC_INLINE void NVIC_SetPriorityGrouping(uint32_t PriorityGroup)
1274     {
1275         uint32_t reg_value;
1276         uint32_t PriorityGroupTmp = (PriorityGroup & (uint32_t)0x07);               /* only values 0..7 are used          */
1277     
1278         reg_value  =  SCB->AIRCR;                                                   /* read old register configuration    */
1279         reg_value &= ~(SCB_AIRCR_VECTKEY_Msk | SCB_AIRCR_PRIGROUP_Msk);             /* clear bits to change               */
1280         reg_value  =  (reg_value                                 |
1281                        ((uint32_t)0x5FA << SCB_AIRCR_VECTKEY_Pos) |
1282                        (PriorityGroupTmp << 8));                                     /* Insert write key and priorty group */
1283         SCB->AIRCR =  reg_value;
1284     }

1287     /** \brief  Get Priority Grouping
1288     
1289       The function reads the priority grouping field from the NVIC Interrupt Controller.
1290     
1291         \return                Priority grouping field (SCB->AIRCR [10:8] PRIGROUP field).
1292      */
1293     __STATIC_INLINE uint32_t NVIC_GetPriorityGrouping(void)
1294     {
1295         return ((SCB->AIRCR & SCB_AIRCR_PRIGROUP_Msk) >> SCB_AIRCR_PRIGROUP_Pos);   /* read priority grouping field */
1296     }

1414         } /* get priority for Cortex-M  system interrupts */
1415         else {

1421     /** \brief  Encode Priority
1422     
1423         The function encodes the priority for an interrupt with the given priority group,
1424         preemptive priority value, and subpriority value.
1425         In case of a conflict between priority grouping and available
1426         priority bits (__NVIC_PRIO_BITS), the smallest possible priority group is set.
1427     
1428         \param [in]     PriorityGroup  Used priority group.
1429         \param [in]   PreemptPriority  Preemptive priority value (starting from 0).
1430         \param [in]       SubPriority  Subpriority value (starting from 0).
1431         \return                        Encoded priority. Value can be used in the function \ref NVIC_SetPriority().
1432      */
1433     __STATIC_INLINE uint32_t NVIC_EncodePriority (uint32_t PriorityGroup, uint32_t PreemptPriority, uint32_t SubPriority)
1434     {
1435         uint32_t PriorityGroupTmp = (PriorityGroup & 0x07);          /* only values 0..7 are used          */
1436         uint32_t PreemptPriorityBits;
1437         uint32_t SubPriorityBits;
1438     
1439         PreemptPriorityBits = ((7 - PriorityGroupTmp) > __NVIC_PRIO_BITS) ? __NVIC_PRIO_BITS : 7 - PriorityGroupTmp;
1440         SubPriorityBits     = ((PriorityGroupTmp + __NVIC_PRIO_BITS) < 7) ? 0 : PriorityGroupTmp - 7 + __NVIC_PRIO_BITS;
1441     
1442         return (
1443                    ((PreemptPriority & ((1 << (PreemptPriorityBits)) - 1)) << SubPriorityBits) |
1444                    ((SubPriority     & ((1 << (SubPriorityBits    )) - 1)))
1445                );
1446     }

1451         The function decodes an interrupt priority value with a given priority group to
1452         preemptive priority value and subpriority value.
1453         In case of a conflict between priority grouping and available
1454         priority bits (__NVIC_PRIO_BITS) the smallest possible priority group is set.
1455     
1456         \param [in]         Priority   Priority value, which can be retrieved with the function \ref NVIC_GetPriority().
1457         \param [in]     PriorityGroup  Used priority group.
1458         \param [out] pPreemptPriority  Preemptive priority value (starting from 0).
1459         \param [out]     pSubPriority  Subpriority value (starting from 0).
1460      */
1461     __STATIC_INLINE void NVIC_DecodePriority (uint32_t Priority, uint32_t PriorityGroup, uint32_t* pPreemptPriority, uint32_t* pSubPriority)
1462     {
1463         uint32_t PriorityGroupTmp = (PriorityGroup & 0x07);          /* only values 0..7 are used          */
1464         uint32_t PreemptPriorityBits;
1465         uint32_t SubPriorityBits;
1466     
1467         PreemptPriorityBits = ((7 - PriorityGroupTmp) > __NVIC_PRIO_BITS) ? __NVIC_PRIO_BITS : 7 - PriorityGroupTmp;
1468         SubPriorityBits     = ((PriorityGroupTmp + __NVIC_PRIO_BITS) < 7) ? 0 : PriorityGroupTmp - 7 + __NVIC_PRIO_BITS;
1469     
1470         *pPreemptPriority = (Priority >> SubPriorityBits) & ((1 << (PreemptPriorityBits)) - 1);
1471         *pSubPriority     = (Priority                   ) & ((1 << (SubPriorityBits    )) - 1);
1472     }

1476     
1477         The function initiates a system reset request to reset the MCU.
1478      */
1479     __STATIC_INLINE void NVIC_SystemReset(void)
1480     {
1481         __DSB();                                                     /* Ensure all outstanding memory accesses included
1482                                                                       buffered write are completed before reset */
1483         SCB->AIRCR  = ((0x5FA << SCB_AIRCR_VECTKEY_Pos)      |
1484                        (SCB->AIRCR & SCB_AIRCR_PRIGROUP_Msk) |
1485                        SCB_AIRCR_SYSRESETREQ_Msk);                   /* Keep priority group unchanged */
1486         __DSB();                                                     /* Ensure completion of memory access */
1487         while(1);                                                    /* wait until reset */
1488     }
